1"""The shared seam every supervisor surface builds on (DRY consolidation, Fix #14). 

2 

3``lifecycle`` / ``roster`` / ``component`` and ``cli.doctor`` each grew their own 

4copy of four identical building blocks: resolve a per-home REST client, decide 

5whether the workspace REST server answers, the one not-running hint, and the 

6``{"data": [...]}``-vs-bare-list process-list normalizer. Four copies are four 

7chances to drift (a different hint string, a different wire-shape tolerance), so 

8they live here once and the surfaces re-export thin aliases for the names their 

9tests reference. 

10 

11Since Phase 2 (the roster moved OFF Process Compose — see :mod:`procspawn`), this 

12module ALSO owns the shared roster-slot primitives: the home-derived shim 

13launcher, the pidfile-namespace scan of ``state/run/*.pid``, and the 

14spawn/terminate glue over :mod:`procspawn` that ``roster`` / ``component`` / 

15``mcp_roster`` all drive. The slot conventions (agent name, ``tools``, 

16``mcp-<server>``) stay with their surfaces; what lives here is the mechanism- 

17agnostic "which slots are alive here" + "spawn/kill this slot" seam, so the four 

18surfaces cannot drift on how a roster member clocks in, out, or is reconciled. 

19 

20Import-light like the rest of :mod:`calfcord.supervisor`, so every CLI-side 

21surface that consumes it stays cheap to import. 

22"""

107async def legacy_pc_roster(client: ProcessComposeClient) -> bool: 

108 """Whether the answering supervisor still supervises ROSTER processes (old main). 

109 

110 ``disco self update`` does not stop a running workspace, so after an upgrade the 

111 old supervisor can still be running the roster as PC-declared slots while 

112 ``state/run/`` is empty. A detached spawn beside that would DUPLICATE the agent 

113 (split-brain: double replies in Discord), so the spawn verbs consult this and 

114 refuse with :data:`LEGACY_WORKSPACE_HINT`; stops and status stay usable so the 

115 operator can wind the old workspace down. 

116 

117 Cheap (one REST read of the project the caller already reached) and FAIL-OPEN: 

118 any failure to read or parse the process list returns ``False`` — detection is 

119 best-effort and must never become a new way for a healthy start to fail. A 

120 row missing its name is skipped for the same reason. 

121 """ 

122 try: 

123 payload = await client.list_processes() 

124 except Exception: 

125 return False 

126 for item in iter_process_dicts(payload): 

127 name = item.get("name") 

128 if isinstance(name, str) and name not in _SUBSTRATE_PROCESS_NAMES: 

129 return True 

130 return False

171def iter_slot_pidfiles(home: str | os.PathLike[str]) -> Iterator[tuple[str, Path]]: 

172 """Yield ``(slot, pidfile)`` for every ``<slot>.pid`` under ``state/run`` (sorted). 

173 

174 The slot is the filename stem, so it round-trips the naming convention the 

175 surfaces write (agent name, ``tools``, ``mcp-<server>``). A missing ``run`` dir 

176 yields nothing rather than raising, so the scan is safe before the first spawn 

177 — but a run dir that EXISTS and cannot be read raises :class:`SlotScanError` 

178 (roster state unknown ≠ no slots; consumers must surface it, not swallow it). 

179 """ 

180 run_dir = _run_dir(home) 

181 try: 

182 entries = os.listdir(run_dir) 

183 except FileNotFoundError: 

184 return 

185 except OSError as exc: 

186 raise SlotScanError(run_dir, exc) from exc 

187 for name in sorted(entries): 

188 if not name.endswith(".pid"): 188 ↛ 189line 188 didn't jump to line 189 because the condition on line 188 was never true

189 continue 

190 stem = name[: -len(".pid")] 

191 try: 

192 procspawn.require_safe_slot(stem) 

193 except ValueError: 

194 # A stray file whose stem could never be a valid slot (dot-leading 

195 # editor/tmp artifacts, traversal shapes): skip it rather than let a 

196 # nonsense stem flow into the path helpers downstream. 

197 continue 

198 yield stem, run_dir / name

201def slot_is_live(home: str | os.PathLike[str], slot: str) -> bool: 

202 """Whether ``slot``'s pidfile names an ours-and-alive process (the truth of "up here"). 

203 

204 A missing/torn/stale pidfile — or one naming a recycled pid — reads ``False`` 

205 (:func:`procspawn.is_ours_and_alive` is the reuse-proof gate). An 

206 INDETERMINATE identity also reads ``False`` (never acted on as ours); 

207 consumers that must preserve those slots use :func:`slot_identity` / 

208 :func:`indeterminate_slots` instead. 

209 """ 

210 record = procspawn.read_pidfile(procspawn.pidfile_for(home, slot)) 

211 return record is not None and procspawn.is_ours_and_alive(record)

223def classify_slots( 

224 home: str | os.PathLike[str], 

225) -> dict[str, procspawn.Identity | None]: 

226 """ONE snapshot of every ``state/run`` slot: slot → identity verdict. 

227 

228 One directory scan with exactly one pidfile read + one identity read per slot 

229 (each identity read is a ``ps`` exec on darwin, so this bounds the cost at N, 

230 not sets x N). ``None`` means the pidfile exists but carries no usable record 

231 (torn/corrupt). Every set-shaped consumer (:func:`live_slots` / 

232 :func:`dead_slots` / :func:`indeterminate_slots` and the status boards) is a 

233 projection of this ONE snapshot, so a slot dying between two scans can never 

234 appear in two disagreeing sets (the cross-scan TOCTOU). Propagates 

235 :class:`SlotScanError` for an unreadable run dir, like the scan it wraps. 

236 """ 

237 verdicts: dict[str, procspawn.Identity | None] = {} 

238 for slot, pidfile in iter_slot_pidfiles(home): 

239 record = procspawn.read_pidfile(pidfile) 

240 verdicts[slot] = None if record is None else procspawn.identity(record) 

241 return verdicts

283def note_local_survivors(home: str | os.PathLike[str]) -> None: 

284 """Note any detached roster processes still alive locally (supervisor down). 

285 

286 Shared by the ``agent ps`` and ``disco status`` early returns: "workspace not 

287 running" alone would imply an idle host, but the detached roster outlives a

346# The post-spawn liveness-confirmation window: long enough for a crash-on-boot 

347# (bad flag, import error, unreadable config) to have exited, short enough that 

348# every start verb stays snappy. Presence/registration is NOT confirmed here — 

349# that is the callers' watchers' job; this only rejects the lie of printing 

350# success for a process that is already dead. 

351_SPAWN_CONFIRM_WINDOW_S = 1.5 

352_SPAWN_CONFIRM_POLL_S = 0.25 

353 

354# Injectable time seams for :func:`launch_slot`, mirroring lifecycle's pattern. 

355Clock = Callable[[], float] 

356Sleep = Callable[[float], Awaitable[None]] 

357 

358 

359async def launch_slot( 

360 home: str | os.PathLike[str], 

361 slot: str, 

362 argv: Sequence[str], 

363 *, 

364 clock: Clock = time.monotonic, 

365 sleep: Sleep = asyncio.sleep, 

366) -> bool: 

367 """Spawn ``slot`` and confirm it survives a short grace window. 

368 

369 Returns ``True`` when the process is still ours-and-alive at the end of the 

370 window ("started" — nothing more). Returns ``False`` only when it PROVABLY 

371 exited during the window (crash-on-boot): the fresh-but-dead pidfile is 

372 cleaned so the failed start leaves no stale record, and the caller reports 

373 the failure with the slot's log path. The check reads the TRI-STATE identity, 

374 not the collapsed boolean: an ``INDETERMINATE`` reading mid-window (a 

375 transient darwin ``ps`` flake) is "keep polling", never "exited immediately" 

376 — and one that persists to the deadline degrades to succeed-with-a-warning 

377 (naming the slot's log), because sweeping the pidfile of a process that may 

378 well be alive and ours would orphan it from every stop/status path. 

379 ``clock``/``sleep`` are injected so tests drive the window without real time. 

380 """ 

381 spawned = spawn_slot(home, slot, argv) 

382 deadline = clock() + _SPAWN_CONFIRM_WINDOW_S 

383 while True: 

384 # The just-spawned process is OUR child and nothing else waits on it, so a 

385 # crash-on-boot leaves a zombie: it still answers ``os.kill(pid, 0)`` and 

386 # (on Linux) keeps its /proc start-token, so without a reap the liveness 

387 # check below would read True for the whole window and the caller would 

388 # print "started" for a dead process. 

389 procspawn.reap(spawned.pid) 

390 verdict = slot_identity(home, slot) 

391 if verdict not in (procspawn.Identity.OURS, procspawn.Identity.INDETERMINATE): 

392 # Provably dead (or a torn/superseded record): the honest crash-on-boot. 

393 procspawn.cleanup_stale(procspawn.pidfile_for(home, slot)) 

394 return False 

395 if clock() >= deadline: 

396 if verdict is procspawn.Identity.INDETERMINATE: 

397 print( 

398 f"warning: {slot} started but its process identity could not " 

399 "be re-read during the confirmation window; assuming it is " 

400 f"running — see {procspawn.log_path_for(home, slot)}" 

401 ) 

402 return True 

403 await sleep(_SPAWN_CONFIRM_POLL_S)

417@contextlib.contextmanager 

418def _flock_nb(path: str | os.PathLike[str], operation: int) -> Iterator[None]: 

419 """Hold a non-blocking ``flock`` on ``path``. 

420 

421 CONTENTION — the lock is held elsewhere — surfaces as ``BlockingIOError`` 

422 (the ``EWOULDBLOCK``/``EAGAIN`` the kernel returns for ``LOCK_NB``; both 

423 supported platforms' ``flock`` reports contention that way and Python maps 

424 those errnos to ``BlockingIOError``). Callers translate exactly that to 

425 their domain busy-error. Any OTHER ``OSError`` — an unmakeable parent dir, an 

426 unwritable lock file — propagates untouched: a permissions problem must read 

427 as a permissions problem, never as "another disco command is in progress".""" 

428 path = os.fspath(path) 

429 os.makedirs(os.path.dirname(path), exist_ok=True) 

430 fd = os.open(path, os.O_CREAT | os.O_RDWR, 0o644) 

431 try: 

432 fcntl.flock(fd, operation | fcntl.LOCK_NB) 

433 try: 

434 yield 

435 finally: 

436 fcntl.flock(fd, fcntl.LOCK_UN) 

437 finally: 

438 os.close(fd)

441@contextlib.contextmanager 

442def slot_mutation(home: str | os.PathLike[str], slot: str) -> Iterator[None]: 

443 """The spawn critical section for one slot: check-alive → spawn → confirm. 

444 

445 Two locks, ALWAYS in this order (the documented ordering — ``disco start``/ 

446 ``stop`` take only the lifecycle lock, so no cycle is possible): 

447 

448 1. the **lifecycle lock, SHARED** (the same ``state/calfcord-lifecycle.lock`` 

449 ``disco stop`` takes exclusively for its roster sweep) — so a spawn can 

450 never land *behind* a concurrent stop sweep, and a stop can never sweep 

451 mid-spawn. Concurrent spawns share it freely. Contention (a start/stop in 

452 flight) raises :class:`WorkspaceBusyError`. 

453 2. an **exclusive per-slot lock** (``state/run/<slot>.lock``) — so two 

454 concurrent ``agent start <name>`` cannot both pass the check-alive and 

455 double-spawn. Contention raises :class:`SlotBusyError`. 

456 

457 Both are non-blocking: a second holder fails immediately with a clear domain 

458 error rather than queueing behind a window it cannot see. 

459 """ 

460 # The per-slot lock path embeds the slot name (and _flock_nb makedirs its 

461 # parent), so a traversal-shaped name must be refused before any path exists. 

462 procspawn.require_safe_slot(slot) 

463 # Lazy import: lifecycle imports this leaf module at module level, so the 

464 # lock-path constant is fetched at call time to avoid an import cycle. 

465 from calfcord.supervisor.lifecycle import _lock_path 

466 

467 try: 

468 lifecycle_guard = _flock_nb(_lock_path(home), fcntl.LOCK_SH) 

469 lifecycle_guard.__enter__() 

470 except BlockingIOError as exc: 

471 # ONLY contention reads as busy; any other OSError (permissions, a broken 

472 # state dir) propagates as the IO problem it actually is. 

473 raise WorkspaceBusyError( 

474 "a `disco start`/`disco stop` is in progress for this workspace; retry once it finishes." 

475 ) from exc 

476 try: 

477 try: 

478 # The run dir is created HERE (owner-only: pidfiles are trusted input 

479 # to a kill path) rather than left to _flock_nb's default-mode makedirs. 

480 procspawn.ensure_private_dir(_run_dir(home)) 

481 slot_guard = _flock_nb(_run_dir(home) / f"{slot}.lock", fcntl.LOCK_EX) 

482 slot_guard.__enter__() 

483 except BlockingIOError as exc: 

484 raise SlotBusyError( 

485 f"another disco command is already starting/stopping {slot!r} here." 

486 ) from exc 

487 try: 

488 yield 

489 finally: 

490 slot_guard.__exit__(None, None, None) 

491 finally: 

492 lifecycle_guard.__exit__(None, None, None)

510def screen_mesh_names(names: Iterable[str]) -> set[str]: 

511 """Keep only well-formed agent ids from a MESH-derived name list. 

512 

513 The mesh roster is broker-wide input: any producer that can write 

514 ``calf.agents`` chooses its own name, so a name is untrusted bytes until it 

515 matches :data:`AGENT_ID_PATTERN` — printing a non-conforming one verbatim 

516 would hand terminal escape sequences to the status boards. Non-conforming 

517 names are DROPPED with one aggregate warning (never echoed raw, not even in 

518 the warning). The shared screen for every board that renders mesh names 

519 (``disco status`` / ``agent ps``). 

520 """ 

521 # Lazy: the identifier module is a stdlib-only leaf, but importing it runs 

522 # the heavy ``calfcord.agents`` package __init__ (which pulls calfkit /

583async def terminate_slot( 

584 home: str | os.PathLike[str], slot: str 

585) -> procspawn.TerminateResult | None: 

586 """Stop ``slot``'s process and clear its pidfile; ``None`` if there was no record. 

587 

588 Reads the pidfile, terminates the process group it names (SIGTERM→SIGKILL via 

589 :func:`procspawn.terminate`), then removes the now-dead pidfile 

590 (:func:`procspawn.cleanup_stale` keeps only a still-alive one, so a 

591 ``NOT_OURS`` stale file is swept too). Returns the terminate outcome, or ``None`` 

592 when there was no usable record (nothing spawned here, or a torn file — the 

593 torn case is swept with a printed warning, since whatever process it named may 

594 survive the removal unseen). 

595 

596 Two outcomes deliberately KEEP the pidfile: ``INDETERMINATE`` (identity 

597 unreadable — never signal, never unlink; a warning tells the operator the 

598 slot is unverifiable) and ``KILL_UNCONFIRMED`` (the process shrugged off 

599 SIGKILL within the reap window; its pidfile stays so it remains visible to 

600 status/stop rather than becoming an invisible survivor). 

601 """ 

602 pidfile = procspawn.pidfile_for(home, slot) 

603 record = procspawn.read_pidfile(pidfile) 

604 if record is None: 

605 # No usable record: clear a torn file if present, report "nothing here". 

606 if procspawn.cleanup_stale(pidfile): 

607 print( 

608 f"warning: removed unreadable pidfile for slot {slot!r}; " 

609 "the process it named (if any) may still be running — check ps/logs." 

610 ) 

611 return None 

612 result = await procspawn.terminate(record) 

613 if result is procspawn.TerminateResult.INDETERMINATE: 

614 print( 

615 f"warning: cannot verify process {record.pid} for slot {slot!r} " 

616 "(identity read failed); leaving it untouched." 

617 ) 

618 return result 

619 procspawn.cleanup_stale(pidfile) 

620 return result
