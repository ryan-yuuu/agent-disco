1"""The bridge's per-``!mention`` orchestration (spec §5.2). 

2 

3Replaces the old publish-to-Kafka → outbox-consumer round trip with the calfkit 

4caller surface. For each ``!mention`` the handler: 

5 

61. resolves the target against the live mesh roster (R-A2 fail-fast); 

72. starts the agent by name on the caller surface (``client.agent(name).start``); 

83. drains the run's ``stream()`` — splitting native A2A activity (consults + 

9 handoffs) from live progress via the stateful :class:`A2ADispatcher`; 

104. awaits the terminal ``result()`` and posts it under the **responding** agent's 

11 persona (emitter-driven, so a handoff posts the peer's persona for free). 

12 

13The collaborators (history, overrides, the A2A projector, the progress renderer, 

14the reply poster) are injected so this orchestration is unit-testable against a 

15``FakeHandle`` with no Kafka or Discord. 

16"""

49# Notice budget: Discord caps a message at 2000 chars. The cap below is a SOFT 

50# target — the "and N more" elision line (when appended) lives in the 100-char 

51# headroom, so a worst-case notice lands near ~1950, never past 2000. The full 

52# detail is in the log regardless. 

53_DISCORD_NOTICE_LIMIT = 2000 

54_MAX_NOTICE_CHARS = _DISCORD_NOTICE_LIMIT - 100 

55# Per-cause truncation: a single WebFetchError message can run long (full URL + 

56# the echoed URL); cap each line so one verbose cause can't crowd out the others. 

57_MAX_CAUSE_MSG_CHARS = 240

65def _root_cause_failures(report: ErrorReport | None) -> list[ErrorReport]: 

66 """The outermost report per independent failure path in a fault_group. 

67 

68 This is a DELIBERATE narrowing of ``ErrorReport.walk()``. The framework's 

69 ``walk()`` yields every nested report — including ``__cause__`` chain links 

70 (a ``WebFetchError`` wrapping an ``httpx.ConnectError`` yields both), which 

71 would double-count failures in the notice. Calfkit provides no 

72 "sibling-failures-only" traversal (no ``walk_leaves()`` helper exists; the 

73 ``causes`` field serves double duty as both group children and ``__cause__`` 

74 links), so we branch on ``error_type == FaultTypes.FAULT_GROUP`` — the stable 

75 code ``_build_fault_group`` (calfkit ``base.py:1441``) is the sole producer 

76 of, and whose ``causes`` are by construction independent failure paths. 

77 

78 Within a group, EVERY direct child is surfaced (whether it carries a 

79 harvested ``exception`` slot or is itself a minted typed fault like 

80 ``billing.quota_exceeded``) — a mixed group must not silently drop its 

81 typed-fault children. Nested groups are recursed. A top-level non-group fault 

82 surfaces itself only if it carries an ``exception`` slot; minted/framework 

83 faults at the top level yield nothing and the caller falls back to 

84 ``report.message``. 

85 """ 

86 if report is None: 86 ↛ 87line 86 didn't jump to line 87 because the condition on line 86 was never true

87 return [] 

88 if report.error_type == FaultTypes.FAULT_GROUP: 

89 failures: list[ErrorReport] = [] 

90 for child in report.causes: 

91 if child.error_type == FaultTypes.FAULT_GROUP: 

92 failures.extend(_root_cause_failures(child)) 

93 else: 

94 failures.append(child) 

95 return failures 

96 if report.exception is not None: 

97 return [report] 

98 return []

101def _format_cause_line(failure: ErrorReport) -> str: 

102 """One Discord notice line for a root-cause failure: origin, type, and message. 

103 

104 The origin + type prefix the message so a multi-cause notice stays scannable 

105 even when (as is common) several failures share a type (``web_fetch: 

106 WebFetchError — ...``). The type falls back to ``error_type`` for minted 

107 typed faults (``billing.quota_exceeded``) that carry no harvested exception 

108 slot. The message is truncated per-leaf so one verbose cause (a long URL) 

109 can't crowd out the rest. 

110 """ 

111 origin = failure.origin_node_id or "unknown" 

112 exc_type = failure.exception.type if failure.exception is not None else failure.error_type 

113 message = (failure.message or "")[:_MAX_CAUSE_MSG_CHARS] 

114 return f" • {origin}: {exc_type} — {message}"

117def _agent_error_text(target: str | None, report: ErrorReport | None = None) -> str: 

118 """Build the user-facing fault notice, surfacing root-cause exceptions. 

119 

120 Surfaces the outermost exception per failure path so the user can tell "4 

121 fetches 403'd" from "agent crashed" — without this, a fault_group's children 

122 are invisible (they live in ``report.causes``, which the old origin-only 

123 notice never read). When there are no failures to surface (a minted/framework 

124 fault like ``billing.quota_exceeded``), the notice includes ``report.message`` 

125 if it adds actionable context, else falls back to the honest generic form. 

126 

127 Leak posture: ``failure.message`` is ``safe_exc_message(exc)`` — the raw 

128 exception message the failing tool raised. It is posted to Discord on the 

129 TRUST assumption that the agent's own tool exceptions (the only producers in 

130 this system) keep their messages free of secrets. ``exception.attrs`` (which 

131 may carry sanitized ``vars(exc)`` like status codes) is deliberately kept 

132 log-only, never posted. 

133 """ 

134 who = f"`{target}`" if target else "The agent" 

135 header = f"{who} hit an error handling that message:" 

136 failures = _root_cause_failures(report) 

137 

138 if not failures: 

139 # No harvested exception to surface. Include the report's message if it's 

140 # informative (a minted fault's message often is — e.g. "quota exhausted"); 

141 # otherwise stay generic rather than expose a raw framework code. 

142 message = report.message if report is not None else None 

143 if message: 

144 return f"{header} {message}. Please try again." 

145 return f"{header} Please try again." 

146 

147 footer = "Please try again, or ask an operator to check the logs." 

148 cause_lines = [_format_cause_line(f) for f in failures] 

149 # Soft budget: show as many lines as fit under the cap. The "and N more" 

150 # line (when it appends) lands in the 100-char headroom above the cap. 

151 budget = _MAX_NOTICE_CHARS - len(header) - len(footer) - 2 

152 shown = 0 

153 for line in cause_lines: 

154 if len(line) + 1 > budget: # +1 for the newline joining it 

155 break 

156 budget -= len(line) + 1 

157 shown += 1 

158 lines = [header, *cause_lines[:shown]] 

159 hidden = len(failures) - shown 

160 if hidden: 

161 lines.append(f"… and {hidden} more failure(s) — see the bridge logs.") 

162 lines.append(footer) 

163 return "\n".join(lines)

166def _log_agent_fault(exc: NodeFaultError, target: str) -> None: 

167 """Log the full ``ErrorReport`` calfkit shipped, plus each root cause. 

168 

169 Agents run on other hosts, so the report the bridge received on the fault 

170 (spec §11.1) is the operator's only in-hand diagnostic — log ``error_type``, 

171 ``message``, ``retryable`` and the harvested upstream exception at ERROR, not 

172 just ``origin`` (which alone forces a cross-host log dig for every fault). 

173 Then surface each root-cause FAILURE (one ERROR line per outermost exception) 

174 with its origin/type/message/attrs: a fault_group summarizes ("N unhandled 

175 fault(s)") but the actual failures (the 403s, the exception types) live in 

176 the children — without this the operator sees the count but not the causes. 

177 The defensive ``getattr`` chain on the summary line tolerates a malformed 

178 fault with no report; ``_root_cause_failures`` is null-safe on its own. 

179 """ 

180 report = getattr(exc, "report", None) 

181 origin = getattr(report, "origin_node_id", None) 

182 exception = getattr(report, "exception", None) 

183 upstream = f"{exception.type}: {exception.attrs}" if exception is not None else None 

184 logger.error( 

185 "agent run faulted target=%s origin=%s error_type=%s retryable=%s message=%s upstream=%s", 

186 target, 

187 origin, 

188 getattr(report, "error_type", None), 

189 getattr(report, "retryable", None), 

190 getattr(report, "message", None), 

191 upstream, 

192 ) 

193 # One ERROR line per root-cause failure so each is named individually 

194 # (origin/type/message/attrs) — the group summary alone hides them. A failure 

195 # may lack an ``exception`` slot (a minted typed fault like 

196 # ``billing.quota_exceeded`` in a mixed group); fall back to ``error_type``. 

197 failures = _root_cause_failures(report) 

198 for i, failure in enumerate(failures, 1): 

199 failure_exc = failure.exception 

200 logger.error( 

201 "agent run root cause [%d/%d] origin=%s type=%s message=%s attrs=%s", 

202 i, 

203 len(failures), 

204 failure.origin_node_id, 

205 failure_exc.type if failure_exc is not None else failure.error_type, 

206 failure.message, 

207 failure_exc.attrs if failure_exc is not None else None, 

208 )

305 async def handle(self, req: MentionRequest) -> None: 

306 # Refresh the mesh snapshot once per turn so the (synchronous) online() 

307 # read below reflects the current roster — there is no background refresh 

308 # loop; a mesh read is an in-memory ktable snapshot. 

309 await self._roster.refresh() 

310 online = self._roster.online() 

311 if online is None: 

312 # Mesh unavailable — we cannot tell who is online, so fail fast 

313 # rather than route blindly (R-A2). reader_dead stays here until the 

314 # bridge restarts; the roster already alerted. 

315 await self._reply.post_notice(req, _ROSTER_UNAVAILABLE) 

316 return 

317 target = next((m for m in req.mention_ids if m in online), None) 

318 if target is None: 

319 if req.mention_ids: 

320 if req.route_kind == "sticky": 

321 await self._reply.post_notice( 

322 req, 

323 _STICKY_OWNER_OFFLINE.format(mention=f"{MENTION_PREFIX}{req.mention_ids[0]}"), 

324 ) 

325 return 

326 # Mentioned an agent that is not online right now. 

327 await self._reply.post_notice(req, _none_online_text(req.mention_ids)) 

328 # else: no !mention at all → ambient → unanswered (C2): do nothing. 

329 return 

330 

331 history = await self._history.message_history(req) 

332 # Serialize the typed wire into deps once per turn (the agent reads 

333 # ``deps["discord"]`` as JSON); the reply poster uses ``req.wire`` typed. 

334 deps = {"discord": req.wire.model_dump(mode="json"), **self._memory_deps()} 

335 # The C11 effort override for this turn (provider-blind union). 

336 model_settings = build_model_settings_union(self._overrides.effort_for(target)) 

337 handle = await self._client.agent(target).start( 

338 req.content, 

339 message_history=history, 

340 deps=deps, 

341 author=req.author_label, 

342 model_settings=model_settings, 

343 ) 

344 

345 dispatcher = A2ADispatcher() 

346 # The owning agent is who is currently in control of the run. It starts 

347 # as the mention target and transfers to the peer on each handoff, so 

348 # tool progress lines (tool_call/tool_result) after a handoff are stamped 

349 # with the new agent's persona, not the original target's. 

350 owning_agent = target 

351 try: 

352 async for event in handle.stream(): 

353 try: 

354 step = normalize_run_event(event) 

355 if step is None: 

356 continue # terminal — handled by result() below 

357 # THE invariant, applied uniformly below: only the agent in 

358 # control of the human's turn may touch the human's thread or 

359 # transfer that control. Any other emitter is inside a consulted 

360 # peer's private sub-tree, reaching us only because calfkit 

361 # flushes EVERY hop to the ROOT caller (base.py ``_flush_steps`` 

362 # → ``stack.root.callback_topic``). ``A2ARequest.caller`` IS the 

363 # step's emitter, so this one predicate covers both branches. 

364 is_owner = step.emitter == owning_agent 

365 projection = dispatcher.classify(step) 

366 if projection is not None: 

367 # Audit EVERY consult, nested peer-to-peer ones included — 

368 # the audit channel is the system of record for the whole 

369 # run tree. The url is this render's receipt: None means it 

370 # failed (swallowed), so there is no thread to link. 

371 url = await self._a2a.project(projection) 

372 if isinstance(projection, A2ARequest) and is_owner: 

373 # Cross-link the consult from the human's thread into the 

374 # audit thread now holding it. ONE marker per consult: the 

375 # reply lands in that same thread. Only the OWNER's 

376 # consults earn one — a peer's nested consult is its own 

377 # private business, and marking it here would re-create 

378 # the very leak this marker replaced. 

379 await self._progress.on_note( 

380 render_consult_marker(projection.peer, url), 

381 req, 

382 correlation_id=projection.correlation_id, 

383 persona_name=projection.caller, 

384 ) 

385 elif is_owner: 

386 await self._progress.on_step(step, req, owning_agent=owning_agent) 

387 else: 

388 # A consulted peer's own step (its preamble, its tool calls). 

389 # That is private A2A work — the audit channel's business — 

390 # and rendering it would spill the exchange into the human's 

391 # thread. 

392 logger.debug( 

393 "bridge: not live progress — step from consulted peer emitter=%s owner=%s kind=%s", 

394 step.emitter, 

395 owning_agent, 

396 step.kind, 

397 ) 

398 # A handoff TRANSFERS control, so the new owner's steps keep 

399 # rendering. Update AFTER rendering, so the announcement itself 

400 # stays under the handing-off agent's persona — and only for the 

401 # OWNER's own handoff: a consulted peer handing off inside its 

402 # private sub-tree never had the human's turn to give away, and 

403 # letting it seize ownership blackholes every later owner step. 

404 if step.kind == "handoff" and step.target and is_owner: 

405 owning_agent = step.target.removeprefix("/") 

406 except Exception: 

407 # A render/normalize/classify bug (or a future calfkit event 

408 # shape) must NOT unwind the drain and cost the user the 

409 # already-computed terminal reply — the progress and A2A 

410 # contracts both promise the render path can't fault the turn. 

411 # Drop just this step (logged) and keep draining; _deliver posts 

412 # the terminal reply below regardless. CancelledError is a 

413 # BaseException, so shutdown still propagates. 

414 logger.exception("bridge: dropping unrenderable run step; terminal reply unaffected") 

415 finally: 

416 await self._progress.finish(handle.correlation_id) 

417 

418 await self._deliver(req, handle, dispatcher, target, history)

420 async def _deliver( 

421 self, 

422 req: MentionRequest, 

423 handle: Any, 

424 dispatcher: A2ADispatcher, 

425 target: str, 

426 history: list[ModelMessage], 

427 ) -> None: 

428 """Post the agent's reply as chunked persona messages (single-pass). 

429 

430 The poster chunk-splits the reply and posts every chunk — there is no 

431 retry that re-invokes the agent. ``"posted"`` (≥1 chunk delivered) 

432 sets the sticky owner. ``"lost"`` (every chunk failed — e.g. missing 

433 Manage Webhooks, bad token, persistent 5xx) would otherwise ghost the 

434 user — the progress message was already deleted in ``handle()`` — so 

435 surface an operator notice via the native-reply path, which needs only 

436 Send Messages and is independent of the failing webhook path. 

437 """ 

438 result = await self._await_terminal(req, handle, dispatcher, target) 

439 if result is None: 

440 return # faulted — notice already posted 

441 

442 persona = persona_for(result.emitter_node_id or target) 

443 outcome = await self._reply.post_reply( 

444 req, persona, result, initial_len=len(history), correlation_id=handle.correlation_id 

445 ) 

446 if outcome == "posted": 

447 await self._set_sticky_owner(req, persona.name) 

448 elif outcome == "lost": 

449 await self._reply.post_notice(req, _REPLY_DROPPED)

457 async def _post_fault_notice(self, req: MentionRequest, exc: NodeFaultError, target: str) -> None: 

458 """Log the fault's full report (I-1) and post the user-facing error notice. 

459 

460 Surfaces root-cause exceptions (the 403s behind a fault_group) in BOTH the 

461 log and the notice — pass the whole report so ``_agent_error_text`` can 

462 walk ``report.causes`` rather than read only the top-level origin. 

463 """ 

464 _log_agent_fault(exc, target) 

465 await self._reply.post_notice(req, _agent_error_text(target, getattr(exc, "report", None)))

467 async def _await_terminal( 

468 self, req: MentionRequest, handle: Any, dispatcher: A2ADispatcher, target: str 

469 ) -> Any | None: 

470 """Await the run's terminal, or ``None`` after handling a fault. 

471 

472 No timeout: per spec §5.2 the bridge awaits the terminal unbounded (C5 

473 drops app-side timeout policing; a durable run may legitimately pause). A 

474 genuine peer/agent fault faults the whole run (D-2) — calfkit maps 

475 ``RunFailed`` → :class:`NodeFaultError`; any consult still open never got a 

476 reply, so synthesize an A2A failure note for each, then post a user-facing 

477 error (best-effort persona from the faulting node when the report names it). 

478 """ 

479 try: 

480 return await handle.result() 

481 except NodeFaultError as exc: 

482 for call in dispatcher.dangling(): 

483 await self._a2a.project_fault(call) 

484 await self._post_fault_notice(req, exc, target) 

485 return None
