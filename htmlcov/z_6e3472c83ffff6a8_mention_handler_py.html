1"""The bridge's per-``!mention`` orchestration (spec §5.2). 

2 

3Replaces the old publish-to-Kafka → outbox-consumer round trip with the calfkit 

4caller surface. For each ``!mention`` the handler: 

5 

61. resolves the target against the live mesh roster (R-A2 fail-fast); 

72. starts the agent by name on the caller surface (``client.agent(name).start``); 

83. drains the run's ``stream()`` — splitting native A2A activity (consults + 

9 handoffs) from the step trace via the stateful :class:`A2ADispatcher`; 

104. awaits the terminal ``result()`` and posts it under the **responding** agent's 

11 persona (emitter-driven, so a handoff posts the peer's persona for free). 

12 

13The collaborators (history, overrides, the A2A projector, the step-trace renderer, 

14the reply poster) are injected so this orchestration is unit-testable against a 

15``FakeHandle`` with no Kafka or Discord. 

16"""

59# Notice budget: Discord caps a message at 2000 chars. The cap below is a SOFT 

60# target — the "and N more" elision line (when appended) lives in the 100-char 

61# headroom, so a worst-case notice lands near ~1950, never past 2000. The full 

62# detail is in the log regardless. 

63_DISCORD_NOTICE_LIMIT = 2000 

64_MAX_NOTICE_CHARS = _DISCORD_NOTICE_LIMIT - 100 

65# Per-cause truncation: a single WebFetchError message can run long (full URL + 

66# the echoed URL); cap each line so one verbose cause can't crowd out the others. 

67_MAX_CAUSE_MSG_CHARS = 240

75def _root_cause_failures(report: ErrorReport | None) -> list[ErrorReport]: 

76 """The outermost report per independent failure path in a fault_group. 

77 

78 This is a DELIBERATE narrowing of ``ErrorReport.walk()``. The framework's 

79 ``walk()`` yields every nested report — including ``__cause__`` chain links 

80 (a ``WebFetchError`` wrapping an ``httpx.ConnectError`` yields both), which 

81 would double-count failures in the notice. Calfkit provides no 

82 "sibling-failures-only" traversal (no ``walk_leaves()`` helper exists; the 

83 ``causes`` field serves double duty as both group children and ``__cause__`` 

84 links), so we branch on ``error_type == FaultTypes.FAULT_GROUP`` — the stable 

85 code ``_build_fault_group`` (calfkit ``base.py:1441``) is the sole producer 

86 of, and whose ``causes`` are by construction independent failure paths. 

87 

88 Within a group, EVERY direct child is surfaced (whether it carries a 

89 harvested ``exception`` slot or is itself a minted typed fault like 

90 ``billing.quota_exceeded``) — a mixed group must not silently drop its 

91 typed-fault children. Nested groups are recursed. A top-level non-group fault 

92 surfaces itself only if it carries an ``exception`` slot; minted/framework 

93 faults at the top level yield nothing and the caller falls back to 

94 ``report.message``. 

95 """ 

96 if report is None: 96 ↛ 97line 96 didn't jump to line 97 because the condition on line 96 was never true

97 return [] 

98 if report.error_type == FaultTypes.FAULT_GROUP: 

99 failures: list[ErrorReport] = [] 

100 for child in report.causes: 

101 if child.error_type == FaultTypes.FAULT_GROUP: 

102 failures.extend(_root_cause_failures(child)) 

103 else: 

104 failures.append(child) 

105 return failures 

106 if report.exception is not None: 

107 return [report] 

108 return []

111def _format_cause_line(failure: ErrorReport) -> str: 

112 """One Discord notice line for a root-cause failure: origin, type, and message. 

113 

114 The origin + type prefix the message so a multi-cause notice stays scannable 

115 even when (as is common) several failures share a type (``web_fetch: 

116 WebFetchError — ...``). The type falls back to ``error_type`` for minted 

117 typed faults (``billing.quota_exceeded``) that carry no harvested exception 

118 slot. The message is truncated per-leaf so one verbose cause (a long URL) 

119 can't crowd out the rest. 

120 """ 

121 origin = failure.origin_node_id or "unknown" 

122 exc_type = failure.exception.type if failure.exception is not None else failure.error_type 

123 message = (failure.message or "")[:_MAX_CAUSE_MSG_CHARS] 

124 return f" • {origin}: {exc_type} — {message}"

127def _agent_error_text(target: str | None, report: ErrorReport | None = None) -> str: 

128 """Build the user-facing fault notice, surfacing root-cause exceptions. 

129 

130 Surfaces the outermost exception per failure path so the user can tell "4 

131 fetches 403'd" from "agent crashed" — without this, a fault_group's children 

132 are invisible (they live in ``report.causes``, which the old origin-only 

133 notice never read). When there are no failures to surface (a minted/framework 

134 fault like ``billing.quota_exceeded``), the notice includes ``report.message`` 

135 if it adds actionable context, else falls back to the honest generic form. 

136 

137 Leak posture: ``failure.message`` is ``safe_exc_message(exc)`` — the raw 

138 exception message the failing tool raised. It is posted to Discord on the 

139 TRUST assumption that the agent's own tool exceptions (the only producers in 

140 this system) keep their messages free of secrets. ``exception.attrs`` (which 

141 may carry sanitized ``vars(exc)`` like status codes) is deliberately kept 

142 log-only, never posted. 

143 """ 

144 who = f"`{target}`" if target else "The agent" 

145 header = f"{who} hit an error handling that message:" 

146 failures = _root_cause_failures(report) 

147 

148 if not failures: 

149 # No harvested exception to surface. Include the report's message if it's 

150 # informative (a minted fault's message often is — e.g. "quota exhausted"); 

151 # otherwise stay generic rather than expose a raw framework code. 

152 message = report.message if report is not None else None 

153 if message: 

154 return f"{header} {message}. Please try again." 

155 return f"{header} Please try again." 

156 

157 footer = "Please try again, or ask an operator to check the logs." 

158 cause_lines = [_format_cause_line(f) for f in failures] 

159 # Soft budget: show as many lines as fit under the cap. The "and N more" 

160 # line (when it appends) lands in the 100-char headroom above the cap. 

161 budget = _MAX_NOTICE_CHARS - len(header) - len(footer) - 2 

162 shown = 0 

163 for line in cause_lines: 

164 if len(line) + 1 > budget: # +1 for the newline joining it 

165 break 

166 budget -= len(line) + 1 

167 shown += 1 

168 lines = [header, *cause_lines[:shown]] 

169 hidden = len(failures) - shown 

170 if hidden: 

171 lines.append(f"… and {hidden} more failure(s) — see the bridge logs.") 

172 lines.append(footer) 

173 return "\n".join(lines)

176def _log_agent_fault(exc: NodeFaultError, target: str) -> None: 

177 """Log the full ``ErrorReport`` calfkit shipped, plus each root cause. 

178 

179 Agents run on other hosts, so the report the bridge received on the fault 

180 (spec §11.1) is the operator's only in-hand diagnostic — log ``error_type``, 

181 ``message``, ``retryable`` and the harvested upstream exception at ERROR, not 

182 just ``origin`` (which alone forces a cross-host log dig for every fault). 

183 Then surface each root-cause FAILURE (one ERROR line per outermost exception) 

184 with its origin/type/message/attrs: a fault_group summarizes ("N unhandled 

185 fault(s)") but the actual failures (the 403s, the exception types) live in 

186 the children — without this the operator sees the count but not the causes. 

187 The defensive ``getattr`` chain on the summary line tolerates a malformed 

188 fault with no report; ``_root_cause_failures`` is null-safe on its own. 

189 """ 

190 report = getattr(exc, "report", None) 

191 origin = getattr(report, "origin_node_id", None) 

192 exception = getattr(report, "exception", None) 

193 upstream = f"{exception.type}: {exception.attrs}" if exception is not None else None 

194 logger.error( 

195 "agent run faulted target=%s origin=%s error_type=%s retryable=%s message=%s upstream=%s", 

196 target, 

197 origin, 

198 getattr(report, "error_type", None), 

199 getattr(report, "retryable", None), 

200 getattr(report, "message", None), 

201 upstream, 

202 ) 

203 # One ERROR line per root-cause failure so each is named individually 

204 # (origin/type/message/attrs) — the group summary alone hides them. A failure 

205 # may lack an ``exception`` slot (a minted typed fault like 

206 # ``billing.quota_exceeded`` in a mixed group); fall back to ``error_type``. 

207 failures = _root_cause_failures(report) 

208 for i, failure in enumerate(failures, 1): 

209 failure_exc = failure.exception 

210 logger.error( 

211 "agent run root cause [%d/%d] origin=%s type=%s message=%s attrs=%s", 

212 i, 

213 len(failures), 

214 failure.origin_node_id, 

215 failure_exc.type if failure_exc is not None else failure.error_type, 

216 failure.message, 

217 failure_exc.attrs if failure_exc is not None else None, 

218 )

360 async def handle(self, req: MentionRequest) -> None: 

361 # Refresh the mesh snapshot once per turn so the (synchronous) online() 

362 # read below reflects the current roster — there is no background refresh 

363 # loop; a mesh read is an in-memory ktable snapshot. 

364 await self._roster.refresh() 

365 online = self._roster.online() 

366 if online is None: 

367 # Mesh unavailable — we cannot tell who is online, so fail fast 

368 # rather than route blindly (R-A2). reader_dead stays here until the 

369 # bridge restarts; the roster already alerted. 

370 await self._reply.post_notice(req, _ROSTER_UNAVAILABLE) 

371 return 

372 target = next((m for m in req.mention_ids if m in online), None) 

373 if target is None: 

374 if req.mention_ids: 

375 if req.route_kind == "sticky": 

376 await self._reply.post_notice( 

377 req, 

378 _STICKY_OWNER_OFFLINE.format(mention=f"{MENTION_PREFIX}{req.mention_ids[0]}"), 

379 ) 

380 return 

381 # Mentioned an agent that is not online right now. 

382 await self._reply.post_notice(req, _none_online_text(req.mention_ids)) 

383 # else: no !mention at all → ambient → unanswered (C2): do nothing. 

384 return 

385 

386 history = await self._history.message_history(req) 

387 # Serialize the typed wire into deps once per turn (the agent reads 

388 # ``deps["discord"]`` as JSON); the reply poster uses ``req.wire`` typed. 

389 deps = {"discord": req.wire.model_dump(mode="json"), **self._memory_deps()} 

390 # The C11 effort override for this turn (provider-blind union). 

391 model_settings = build_model_settings_union(self._overrides.effort_for(target)) 

392 handle = await self._client.agent(target).start( 

393 req.content, 

394 message_history=history, 

395 deps=deps, 

396 author=req.author_label, 

397 model_settings=model_settings, 

398 ) 

399 dispatcher = A2ADispatcher() 

400 # The acting agent is who is currently in control of the run. It starts 

401 # as the mention target and transfers to the peer on each handoff, so 

402 # tool trace rows (tool_call/tool_result) after a handoff are stamped 

403 # with the new agent's persona, not the original target's. 

404 acting_agent = target 

405 # The human's trace renders where the mention landed: the webhook hosts on 

406 # the flattened parent, and posts into the thread when the wire came from 

407 # one. Fixed for the turn — a handoff changes who speaks, never where. 

408 human_dest = Destination( 

409 channel_id=req.channel_id, 

410 thread_id=(req.source_channel_id if req.source_channel_id != req.channel_id else None), 

411 ) 

412 # The A2A projector is retired in the OUTER finally, after _deliver: a 

413 # faulted run synthesizes its dangling-consult notes during delivery, and 

414 # those need the turn's audit thread still mapped (ADR-0026). Retiring it 

415 # alongside the human trace below would evict the mapping first and orphan 

416 # each note in a freshly created second thread. 

417 try: 

418 await self._a2a.begin_turn( 

419 correlation_id=handle.correlation_id, 

420 root_agent=target, 

421 subject=req.content, 

422 ) 

423 await self._drain_and_deliver(req, handle, dispatcher, target, history, acting_agent, human_dest) 

424 finally: 

425 await self._a2a.finish(handle.correlation_id)

427 async def _drain_and_deliver( 

428 self, 

429 req: MentionRequest, 

430 handle: Any, 

431 dispatcher: A2ADispatcher, 

432 target: str, 

433 history: list[ModelMessage], 

434 acting_agent: str, 

435 human_dest: Destination, 

436 ) -> None: 

437 """Drain the run's stream into both trace surfaces, then post the reply. 

438 

439 Split out so ``handle`` can wrap it in the outer ``finally`` that retires 

440 the A2A projector — which must happen after ``_deliver``, not alongside 

441 the human trace's ``finish``. 

442 """ 

443 try: 

444 async for event in handle.stream(): 

445 try: 

446 step = normalize_run_event(event) 

447 if step is None: 

448 # The stream's terminal. It seals the trace with the run's 

449 # outcome — the ONLY moment the bridge knows it in time, 

450 # since `finish` runs in the `finally` below while the 

451 # fault only surfaces afterwards in `_await_terminal` 

452 # (ADR-0025). `result()` still owns the reply and the 

453 # notice; this is display only, and best-effort. 

454 faulted = normalize_terminal(event) 

455 if faulted is not None: 455 ↛ 467line 455 didn't jump to line 467 because the condition on line 455 was always true

456 # BOTH surfaces seal: a consulted agent's trace ends 

457 # with the same run, and #124's motivating case is 

458 # exactly this — 23 tool calls then a fault, which 

459 # must read "run failed after 23 tools", not trail off. 

460 await self._trace.seal(handle.correlation_id, faulted=faulted) 

461 await self._a2a.seal(handle.correlation_id, faulted=faulted) 

462 continue 

463 # Not a terminal either: an event kind the normalizer does 

464 # not know (a future calfkit step type). Dropping it is 

465 # right, dropping it SILENTLY is not — this is the only 

466 # place the two "None"s are distinguishable. 

467 logger.warning("bridge: unrenderable run event %s; dropping", type(event).__name__) 

468 continue 

469 # THE invariant, applied uniformly below: only the agent in 

470 # control of the human's turn may touch the human's thread or 

471 # transfer that control. Any other emitter is inside a consulted 

472 # peer's private sub-tree, reaching us only because calfkit 

473 # flushes EVERY hop to the ROOT caller (base.py ``_flush_steps`` 

474 # → ``stack.root.callback_topic``). ``A2ARequest.caller`` IS the 

475 # step's emitter, so this one predicate covers both branches. 

476 is_acting = step.emitter == acting_agent 

477 projection = dispatcher.classify(step) 

478 if projection is not None and is_acting: 

479 # The ACTING agent's consult: project the exchange to the 

480 # audit thread and cross-link it with ONE resolving row in 

481 # the human's thread — opened by the request, resolved by 

482 # its outcome, carrying only the peer, state, and link 

483 # (ADR-0020). The url is the render's receipt: None means 

484 # it failed (swallowed), so there is no thread to link. 

485 url = await self._a2a.project(projection) 

486 await self._render_consult(projection, url, human_dest) 

487 elif isinstance(projection, A2ARequest): 

488 # A NESTED consult's request (the peer is not the acting 

489 # agent). Announce it as a resolving row in the caller's 

490 # OWN trace inside the audit thread (ADR-0027), and SUPPRESS 

491 # the standalone prompt message — the row is the single 

492 # signal, not a bare `[caller] <prompt>` line that reads as 

493 # the peer's own words and leaves its work unannounced. 

494 # 

495 # Edge: if a handoff makes this exact caller the acting 

496 # agent BETWEEN its request and its reply, the reply routes 

497 # to the human thread (is_acting) and this audit row goes 

498 # unresolved — the seal then closes it "interrupted". A 

499 # visible, bounded degradation, not a silent drop. 

500 await self._a2a.project_consult(projection) 

501 elif projection is not None: 

502 # Resolve the nested route row BEFORE posting any peer 

503 # response. The response path flushes the trace, so this 

504 # ordering guarantees the row cannot remain pending above 

505 # an answer that has already arrived. 

506 await self._a2a.project_consult_result(projection) 

507 await self._a2a.project(projection) 

508 elif is_acting: 

509 await self._trace.on_step(step, human_dest, acting_agent=acting_agent) 

510 else: 

511 # A consulted agent's own step (its preamble, its tool 

512 # calls). That is private A2A work — the audit channel's 

513 # business — so it renders in the turn's A2A thread under 

514 # that agent's own persona (ADR-0026). Rendering it HERE 

515 # would spill the exchange into the human's conversation, 

516 # which is what the is_acting guard exists to prevent. 

517 await self._a2a.project_step(step) 

518 # A handoff TRANSFERS control, so the new owner's steps keep 

519 # rendering. Update AFTER rendering, so the announcement itself 

520 # stays under the handing-off agent's persona — and only for the 

521 # OWNER's own handoff: a consulted peer handing off inside its 

522 # private sub-tree never had the human's turn to give away, and 

523 # letting it seize ownership blackholes every later owner step. 

524 if step.kind == "handoff" and step.target and is_acting: 

525 acting_agent = step.target.removeprefix("/") 

526 except Exception: 

527 # A render/normalize/classify bug (or a future calfkit event 

528 # shape) must NOT unwind the drain and cost the user the 

529 # already-computed terminal reply — the trace and A2A 

530 # contracts both promise the render path can't fault the turn. 

531 # Drop just this step (logged) and keep draining; _deliver posts 

532 # the terminal reply below regardless. CancelledError is a 

533 # BaseException, so shutdown still propagates. 

534 logger.exception("bridge: dropping unrenderable run step; terminal reply unaffected") 

535 finally: 

536 await self._trace.finish(handle.correlation_id) 

537 

538 await self._deliver(req, handle, dispatcher, target, history) 

539 

540 async def _deliver( 

541 self, 

542 req: MentionRequest, 

543 handle: Any, 

544 dispatcher: A2ADispatcher, 

545 target: str, 

546 history: list[ModelMessage], 

547 ) -> None: 

548 """Post the agent's reply as chunked persona messages (single-pass). 

549 

550 The poster chunk-splits the reply and posts every chunk — there is no 

551 retry that re-invokes the agent. ``"posted"`` (≥1 chunk delivered) 

552 sets the sticky owner. ``"lost"`` (every chunk failed — e.g. missing 

553 Manage Webhooks, bad token, persistent 5xx) would otherwise ghost the 

554 user — the step-trace message was already deleted in ``handle()`` — so 

555 surface an operator notice via the native-reply path, which needs only 

556 Send Messages and is independent of the failing webhook path. 

557 """ 

558 result = await self._await_terminal(req, handle, dispatcher, target) 

559 if result is None: 

560 return # faulted — notice already posted 

561 

562 persona = persona_for(result.emitter_node_id or target) 

563 outcome = await self._reply.post_reply( 

564 req, persona, result, initial_len=len(history), correlation_id=handle.correlation_id 

565 ) 

566 if outcome == "posted": 

567 await self._set_sticky_owner(req, persona.name) 

568 elif outcome == "lost": 

569 await self._reply.post_notice(req, _REPLY_DROPPED)

571 async def _set_sticky_owner(self, req: MentionRequest, owner_agent_id: str) -> None: 

572 """Persist the visible terminal responder as this conversation's owner.""" 

573 if self._sticky is None: 

574 return 

575 await self._sticky.set_sticky_owner(str(req.source_channel_id or req.channel_id), owner_agent_id)

577 async def _render_consult(self, projection: A2AProjection, url: str | None, dest: Destination) -> None: 

578 """Open or resolve the acting agent's consult row in the step trace. 

579 

580 Every projection carries the ``tool_call_id``, so the request opens a 

581 keyed row and each outcome resolves that same row. Previously only the 

582 REQUEST rendered — leaving an optimistic "consulted" line even when the 

583 consult was rejected or faulted, because the ⚠️ went solely to the audit 

584 thread. 

585 """ 

586 if isinstance(projection, A2ARequest): 

587 await self._trace.on_consult( 

588 projection.tool_call_id, 

589 projection.peer, 

590 url, 

591 dest, 

592 correlation_id=projection.correlation_id, 

593 persona_name=projection.caller, 

594 ) 

595 return 

596 # State/note mapping shared with the audit-thread row via consult_outcome 

597 # (ADR-0020's privacy rule — only a rejection's reason surfaces — lives there). 

598 state, note = consult_outcome(projection) 

599 await self._trace.on_consult_result( 

600 projection.tool_call_id, 

601 state=state, 

602 note=note, 

603 correlation_id=projection.correlation_id, 

604 )

607 """Log the fault's full report (I-1) and post the user-facing error notice.

612 """ 

613 _log_agent_fault(exc, target)

616 async def _await_terminal( 

617 self, req: MentionRequest, handle: Any, dispatcher: A2ADispatcher, target: str 

618 ) -> Any | None: 

619 """Await the run's terminal, or ``None`` after handling a fault. 

620 

621 No timeout: per spec §5.2 the bridge awaits the terminal unbounded (C5 

622 drops app-side timeout policing; a durable run may legitimately pause). A 

623 genuine peer/agent fault faults the whole run (D-2) — calfkit maps 

624 ``RunFailed`` → :class:`NodeFaultError`; any consult still open never got a 

625 reply, so synthesize an A2A failure note for each, then post a user-facing 

626 error (best-effort persona from the faulting node when the report names it). 

627 """ 

628 try: 

629 return await handle.result() 

630 except NodeFaultError as exc: 

631 for call in dispatcher.dangling(): 

632 await self._a2a.project_fault(call) 

633 await self._post_fault_notice(req, exc, target) 

634 return None
