1"""Cursor + selection state for the list widgets. 

2 

3Pure: no terminal, no Rich, no I/O. Keeping the logic here means wrap-around 

4navigation, pre-checking, and selection order are unit-testable directly, and 

5:mod:`calfcord.cli.tui.widgets` is left with only rendering and the key loop. 

6""" 

7 

8from __future__ import annotations 

9 

10from calfcord.cli._prompts import Choice 

11 

12# Fallback for direct construction (and tests). Production never reaches it — 

13# the widgets measure the terminal via ``widgets.viewport_for``, whose docstring 

14# explains why any fixed count is wrong in both directions. Kept only so this 

15# module stays standalone-constructible without importing the render layer. 

16DEFAULT_VIEWPORT = 10 

17 

18 

19class ListState: 

20 """Shared scrolling cursor over a non-empty choice list.""" 

21 

22 def __init__(self, choices: list[Choice], *, viewport: int = DEFAULT_VIEWPORT) -> None: 

23 # An empty list renders a prompt with nothing to answer, which would hang 

24 # the operator with no way forward but Ctrl-C. Callers building choices 

25 # dynamically (models fetched from a provider, servers read from 

26 # mcp.json) can produce one, so reject it here — loudly and at 

27 # construction — rather than at paint time. 

28 if not choices: 

29 raise ValueError("a prompt needs at least one choice") 

30 # ``value`` is a primary key: CheckboxState's ``_checked`` is a set of 

31 # values and ``selected`` filters by membership, so two rows sharing a 

32 # value toggle as one and emit that value TWICE — into an agent's 

33 # persisted ``tools:`` list, with a checkmark on a row nobody touched. 

34 # Callers keep duplicates out today (unique registry keys, a set union 

35 # over MCP servers, agent_tools' ``current - offered``), which is exactly 

36 # why this belongs here instead: that care lives far from the type that 

37 # depends on it, and the next choice source would inherit no guardrail. 

38 if len({c.value for c in choices}) != len(choices): 

39 raise ValueError("choice values must be unique") 

40 self.choices = choices 

41 self.cursor = 0 

42 self.viewport = max(1, viewport) 

43 self._offset = 0 

44 

45 def up(self) -> None: 

46 self.cursor = (self.cursor - 1) % len(self.choices) 

47 self._scroll_to_cursor() 

48 

49 def down(self) -> None: 

50 self.cursor = (self.cursor + 1) % len(self.choices) 

51 self._scroll_to_cursor() 

52 

53 def resize(self, viewport: int) -> None: 

54 """Change visible capacity while keeping the active row on screen.""" 

55 self.viewport = max(1, viewport) 

56 self._scroll_to_cursor() 

57 

58 def _scroll_to_cursor(self) -> None: 

59 """Shift the window the minimum needed to keep the cursor inside it. 

60 

61 Minimum-shift rather than re-centring: it keeps the list visually still 

62 while the cursor moves through the middle, which is what makes scanning a 

63 long list feel steady. The two branches also cover the wrap-around jumps 

64 (last→first, first→last), which move the cursor by more than one row and 

65 would otherwise strand the window at the far end of the list. 

66 """ 

67 if self.cursor < self._offset: 

68 self._offset = self.cursor 

69 elif self.cursor >= self._offset + self.viewport: 

70 self._offset = self.cursor - self.viewport + 1 

71 self._offset = max(0, min(self._offset, len(self.choices) - self.viewport)) 

72 

73 def window(self) -> tuple[int, int]: 

74 """The ``(start, stop)`` slice of :attr:`choices` currently on screen. 

75 

76 A list shorter than the viewport returns its whole extent — the window 

77 never pads past the end, so a 2-row prompt draws a 2-row panel rather 

78 than a 10-row one with eight blank lines. 

79 """ 

80 return self._offset, min(self._offset + self.viewport, len(self.choices)) 

81 

82 @property 

83 def scrolled(self) -> bool: 

84 """True when the list is taller than its viewport, so rows are hidden.""" 

85 return len(self.choices) > self.viewport

88class SelectState(ListState): 

89 """Single-choice cursor. 

90 

91 ``default`` names the :attr:`Choice.value` to start on. An unrecognized 

92 default falls back to the first row rather than raising: the value often 

93 comes from stored config that may have gone stale, and a stale value must 

94 not make the command uncallable. 

95 """ 

96 

97 def __init__(self, choices: list[Choice], *, default: str | None = None, viewport: int = DEFAULT_VIEWPORT) -> None: 

98 super().__init__(choices, viewport=viewport) 

99 if default is not None: 

100 self.cursor = next((i for i, c in enumerate(choices) if c.value == default), 0) 

101 # Scroll the default into view at construction: a stored value far 

102 # down a long list (a model slug, an agent name) must be visible when 

103 # the prompt opens, not hidden below the fold with the cursor on it. 

104 self._scroll_to_cursor() 

105 

106 @property 

107 def value(self) -> str: 

108 return self.choices[self.cursor].value

111class CheckboxState(ListState): 

112 """Multi-select cursor, pre-checked from :attr:`Choice.checked`.""" 

113 

114 def __init__(self, choices: list[Choice], *, viewport: int = DEFAULT_VIEWPORT) -> None: 

115 super().__init__(choices, viewport=viewport) 

116 self._checked = {c.value for c in choices if c.checked} 

117 

118 def toggle(self) -> None: 

119 value = self.choices[self.cursor].value 

120 self._checked.symmetric_difference_update({value}) 

121 

122 def is_checked(self, value: str) -> bool: 

123 return value in self._checked 

124 

125 @property 

126 def selected(self) -> list[str]: 

127 """Checked values in CHOICE order, never toggle order. 

128 

129 Callers persist this straight into an agent's ``tools:`` list, so a 

130 stable order keeps the rendered ``.md`` diff clean across edits. 

131 """ 

132 return [c.value for c in self.choices if c.value in self._checked]
