20"""The result of a ``tool_result`` step, mirroring calfkit's ``ToolResultEvent.outcome``. 

21Owned here (not imported) so the renderers depend on THIS seam, never on calfkit's 

22types — the step source can change behind :func:`normalize_run_event` (spec §5.1). 

23``success`` renders as a return; ``failed`` (the call raised / retry-marked) and 

24``denied`` (the caller refused to dispatch) are distinct error kinds — a plain bool 

25would collapse them, so Discord could not render them 1:1."""

60def _render_text(parts: list[Any]) -> str: 

61 """Concatenate the ``TextPart`` text of a step's parts — the human-readable 

62 content the renderers show; non-text parts (files, data) are skipped.""" 

63 return "".join(p.text for p in parts if getattr(p, "kind", None) == "text")

79def normalize_terminal(event: RunEvent) -> bool | None: 

80 """Whether ``event`` is the run's terminal and, if so, whether it FAILED. 

81 

82 ``True`` = failed, ``False`` = completed, ``None`` = not a terminal. The 

83 tri-state matters: ``False`` ("terminal, succeeded") and ``None`` ("not a 

84 terminal") are different answers, so a bare bool would collapse them. 

85 

86 This exists because the step trace must be sealed with the run's outcome, 

87 and the only place that knows it in time is the drain: calfkit's stream is 

88 ``zero or more step events then exactly one terminal``, so the terminal 

89 arrives as the stream's last item — BEFORE the handler's ``finally`` calls 

90 ``finish``. Reading it here needs no control-flow change (ADR-0025). 

91 

92 ``result()`` stays the authority for the reply and the fault notice; the two 

93 cannot disagree, since it derives its ``NodeFaultError`` from this same 

94 terminal. Kept in this module so :func:`normalize_run_event` and this one are 

95 the only code aware of calfkit's event types (spec §5.1). 

96 

97 Discriminated by ``isinstance`` rather than a ``kind`` tag: unlike the step 

98 events, the terminals carry no ``kind`` field. 

99 """ 

100 if isinstance(event, RunFailed): 

101 return True 

102 if isinstance(event, RunCompleted): 

103 return False 

104 return None

107def normalize_run_event(event: RunEvent) -> StepEvent | None: 

108 """Adapt a calfkit ``RunEvent`` into a :class:`StepEvent`, or ``None`` for the 

109 terminals (``RunCompleted`` / ``RunFailed``). 

110 

111 A terminal is :func:`normalize_terminal`'s business — it seals the trace — 

112 and ``handle.result()`` remains the authority for the reply and the fault. 

113 

114 The ONLY code that knows calfkit's step-event types; the renderers and the 

115 A2A dispatcher depend on :class:`StepEvent`, so the transport can change 

116 behind this one seam. 

117 """ 

118 kind = getattr(event, "kind", None) 

119 if kind == "agent_message": 

120 return StepEvent( 

121 kind="agent_message", 

122 correlation_id=event.correlation_id, 

123 depth=event.depth, 

124 emitter=event.emitter, 

125 text=_render_text(event.parts), 

126 ) 

127 if kind == "tool_call": 

128 return StepEvent( 

129 kind="tool_call", 

130 correlation_id=event.correlation_id, 

131 depth=event.depth, 

132 emitter=event.emitter, 

133 tool_call_id=event.tool_call_id, 

134 name=event.name, 

135 args=_args_to_dict(event.args), 

136 ) 

137 if kind == "tool_result": 

138 return StepEvent( 

139 kind="tool_result", 

140 correlation_id=event.correlation_id, 

141 depth=event.depth, 

142 emitter=event.emitter, 

143 tool_call_id=event.tool_call_id, 

144 name=event.name, 

145 text=_render_text(event.parts), 

146 outcome=event.outcome, 

147 ) 

148 if kind == "handoff": 

149 return StepEvent( 

150 kind="handoff", 

151 correlation_id=event.correlation_id, 

152 depth=event.depth, 

153 emitter=event.emitter, 

154 target=event.target, 

155 reason=event.reason, 

156 ) 

157 return None
