1"""Resolve calfcord install-root paths shared by the auth providers. 

2 

3Every provider that persists OAuth credentials (codex, grok, …) keeps them 

4under the install home so they sit beside ``config/.env``, the agents dir, 

5and ``state/`` — and so they move with the install when an operator relocates 

6it (``CALFCORD_HOME``), runs two installs on one host, or runs under systemd. 

7 

8This is the single source of truth for that resolution; per-provider 

9``_paths`` modules re-export from here so the call sites can't drift on the 

10empty-string guard. Every other subsystem (``cli/main.py``, ``mcp/config.py``, 

11``bridge/gateway.py``) roots its paths at ``$CALFCORD_HOME`` the same way. 

12""" 

13 

14from __future__ import annotations 

15 

16import os 

17from pathlib import Path 

18 

19_AUTH_DIR_ENV = "CALFCORD_AUTH_DIR" 

20 

21 

22def calfcord_home() -> Path: 

23 """The install root: ``$CALFCORD_HOME``, else the shim's ``~/.agent-disco`` default. 

24 

25 An empty ``CALFCORD_HOME=`` counts as unset (so a stray assignment doesn't 

26 root paths at ``/``), matching the guard the CLI/mcp/bridge resolvers use. 

27 Resolved at call time, not import, so the env is read where the path is used. 

28 """ 

29 home = os.environ.get("CALFCORD_HOME") 

30 return Path(home) if home else Path.home() / ".agent-disco" 

31 

32 

33def provider_auth_dir() -> Path: 

34 """The shared credential directory for OAuth providers. 

35 

36 ``CALFCORD_AUTH_DIR`` wins (explicit operator intent); otherwise credentials 

37 live beside the rest of the install at ``$CALFCORD_HOME/auth`` (default 

38 ``~/.agent-disco/auth``) so they move with a relocated or per-host install. 

39 An empty ``CALFCORD_AUTH_DIR=`` counts as unset. Files are vendor-namespaced 

40 (``openai_oauth.json`` / ``xai_oauth.json``) so providers coexist here. 

41 """ 

42 override = os.environ.get(_AUTH_DIR_ENV) 

43 return Path(override) if override else calfcord_home() / "auth"
