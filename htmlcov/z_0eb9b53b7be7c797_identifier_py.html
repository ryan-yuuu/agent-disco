1"""Single source of truth for agent_id format constraints. 

2 

3The pattern ``[a-z0-9_-]{1,32}`` appears in several places across the 

4codebase (notably :class:`AgentDefinition.agent_id` and the bridge 

5normalizer's mention scanner). This module is the canonical 

6definition; importers should reach for :data:`AGENT_ID_PATTERN` rather 

7than re-declaring the regex. 

8 

9This module is intentionally a leaf — only stdlib imports — so any 

10package in the codebase can import it without risking cycles. 

11""" 

12 

13from __future__ import annotations 

14 

15import re 

16 

17AGENT_ID_CHARSET = "a-z0-9_-" 

18"""Raw character class (without brackets) for agent_id characters. 

19Use for building related regexes (e.g. the bridge's mention 

20scanner) where the surrounding pattern shape differs but the 

21character set must match.""" 

22 

23MENTION_PREFIX = "!" 

24"""The single character that prefixes an agent mention in a Discord message 

25(e.g. ``!scribe``). The bridge normalizer builds its mention scanner from this 

26constant, so the trigger char has one home and cannot drift between call sites. 

27 

28Deliberately NOT a member of :data:`AGENT_ID_CHARSET`: the prefix must never be 

29a legal agent-id character, or the scanner could not tell where the prefix ends 

30and the id begins.""" 

31 

32_AGENT_ID_REGEX_STR = rf"[{AGENT_ID_CHARSET}]{{1,32}}" 

33 

34AGENT_ID_PATTERN = re.compile(_AGENT_ID_REGEX_STR) 

35"""Compiled regex matching the canonical agent_id format. Use 

36``.fullmatch(value)`` for membership checks; the character class 

37also appears verbatim inside the bridge normalizer's mention scanner.""" 

38 

39RESERVED_AGENT_IDS = frozenset({"broker", "bridge", "tools", "process-compose", "unstick", "new"}) 

40"""Workspace slot names an agent id may never take. 

41 

42Agents share one process/slot namespace with the substrate (the ``broker`` and 

43``bridge`` Process Compose processes) and the ``tools`` singleton (a detached 

44roster slot under ``state/run/<slot>.pid``); an agent named after any of them 

45would collide with that process. ``process-compose`` is reserved for a LOG 

46collision instead of a slot one: every roster slot logs to 

47``state/logs/<slot>.log``, and the supervisor's own ``-L`` log is 

48``state/logs/process-compose.log`` — an agent by that name would interleave with 

49it, and rotate-at-spawn would rename the live supervisor's log out from under 

50it. Canonically homed here (the leaf id module) so the parse-time validator 

51(:class:`~calfcord.agents.definition.AgentDefinition`) and the supervisor's slot 

52set (:mod:`calfcord.supervisor.compose`) import ONE definition and cannot 

53drift.""" 

54 

55MCP_SLOT_PREFIX = "mcp-" 

56"""The slot-name prefix reserved for MCP servers (``mcp-<server>``). 

57 

58Shares this module with :data:`RESERVED_AGENT_IDS` for the same reason: agent 

59ids and MCP slots live in the same pidfile namespace, so the prefix is both the 

60supervisor's slot convention (re-exported by :mod:`calfcord.supervisor.compose`) 

61and a reserved prefix for agent names.""" 

62 

63 

64def reserved_agent_id_error(agent_id: str) -> str | None: 

65 """Why ``agent_id`` is a reserved workspace-slot name, or ``None`` if it's free. 

66 

67 The one create-/parse-time chokepoint message: both ``disco agent create``'s 

68 name prompt and :class:`AgentDefinition`'s validator surface exactly this 

69 text, so the operator hears the same reason wherever the collision is caught. 

70 (The roster verbs keep their own runtime refusal as defense-in-depth.) 

71 """ 

72 if agent_id == "process-compose": 

73 # The collision here is the LOG file, not a process slot: the supervisor's 

74 # own log is state/logs/process-compose.log, exactly where an agent by 

75 # this name would write (and rotate at spawn). 

76 return ( 

77 "'process-compose' is reserved for the workspace supervisor — an " 

78 "agent by this name would share (and rotate) the supervisor's log " 

79 "file; pick another agent name" 

80 ) 

81 if agent_id == "unstick": 

82 return ( 

83 "'unstick' is reserved for the Discord !unstick routing command — " 

84 "pick another agent name" 

85 ) 

86 if agent_id == "new": 

87 return ( 

88 "'new' is reserved for the Discord !new routing command — " 

89 "pick another agent name" 

90 ) 

91 if agent_id in RESERVED_AGENT_IDS: 

92 role = "substrate" if agent_id in ("broker", "bridge") else "tools host" 

93 return ( 

94 f"{agent_id!r} is reserved for the workspace's {role} process — " 

95 "agents share its process namespace; pick another agent name" 

96 ) 

97 if agent_id.startswith(MCP_SLOT_PREFIX): 

98 return ( 

99 f"{agent_id!r} starts with 'mcp-', which is reserved for MCP server " 

100 "slots (mcp-<server>) in the workspace's process namespace; " 

101 "pick another agent name" 

102 ) 

103 return None
