1"""Bridge-hosted, read-only Discord tool nodes. 

2 

3These nodes are built dynamically around the bridge's authenticated 

4:class:`~calfcord.discord.read_service.DiscordReadService`. They are NOT part 

5of :mod:`calfcord.tools`' generic registry: only the Discord bridge receives the 

6bot token, and only its co-located Worker may advertise this surface. 

7"""

20def build_discord_tool_nodes(service: DiscordReadService) -> list[ToolNodeDef]: 

21 """Build the two read-only tool nodes bound to ``service``.""" 

22 

23 async def discord_list_channels() -> dict[str, Any]: 

24 """List Discord channels and active threads visible to the bot. 

25 

26 Returns channel IDs, names, types, categories, parent relationships, and 

27 whether message history is readable. The server is fixed by the bridge 

28 configuration; this tool cannot inspect another server. 

29 """ 

30 return await service.list_channels() 

31 

32 async def discord_read_messages( 

33 channel_id: int, 

34 limit: int = 50, 

35 before_message_id: int | None = None, 

36 after_message_id: int | None = None, 

37 ) -> dict[str, Any]: 

38 """Read recent messages from one visible Discord channel or thread. 

39 

40 Args: 

41 channel_id: Channel or thread ID returned by discord_list_channels. 

42 limit: Number of messages to return, from 1 to 100 (default 50). 

43 before_message_id: Return messages older than this message ID. 

44 after_message_id: Return messages newer than this message ID. Do not 

45 combine with before_message_id. 

46 """ 

47 return await service.read_messages( 

48 channel_id, 

49 limit=limit, 

50 before_message_id=before_message_id, 

51 after_message_id=after_message_id, 

52 ) 

53 

54 return [ 

55 agent_tool(discord_list_channels), 

56 agent_tool(discord_read_messages), 

57 ]
