1"""Output helpers for the interactive flows. 

2 

3Two rules make it safe to route existing operator prose through Rich, and it is 

4worth being precise about which mechanism enforces which: 

5 

6* **The prose is wrapped in ``Text()`` before printing.** *That* — not 

7 ``markup=False`` — is what keeps bracketed text and ``$``-sigils literal, because 

8 Rich parses markup only when converting a ``str`` into a Text. Delete the 

9 ``Text()`` wrapper and the brackets get eaten as style tags no matter what the 

10 flags say. ``markup=False``/``highlight=False`` are passed as belt-and-braces so 

11 the contract survives someone later handing :func:`line` a bare string. 

12* ``soft_wrap=True`` — disables wrapping and cropping, so a line reaches the 

13 terminal byte-identical to what ``print`` would have emitted. Rich's default 

14 is to wrap at the console width (80 off-TTY), which would bisect the phrases 

15 the CLI's existing tests match on. This one is genuinely load-bearing. 

16 

17Both are passed **per call**, never set on the console. As constructor defaults 

18they would apply to every render — including the widgets', where ``soft_wrap`` 

19propagates into a Panel's children and cuts long choice labels at the panel edge 

20instead of wrapping them. Prose wants no wrapping; a menu row wants wrapping. The 

21console must not decide for both. 

22 

23``highlight=False`` is part of the same contract: Rich's automatic highlighter 

24would otherwise colour numbers, paths, and URLs inside plain sentences. 

25 

26:func:`answer` and :func:`header` bypass :func:`line` and print directly. They are 

27safe for a *different* reason — they build ``Text`` objects, which are never 

28markup-parsed — not because they funnel. A future helper that prints a raw ``str`` 

29must use :func:`line`, or it inherits the markup-eating bug these rules exist to 

30prevent. 

31"""

50def make_console(*, width: int | None = None, record: bool = False) -> Console: 

51 """Build a console. ``record`` + ``width`` are for tests; production uses neither.""" 

52 # soft_wrap is NOT set here. As a constructor default it sets no_wrap + 

53 # overflow="ignore" for EVERY render, and those options propagate into a 

54 # Panel's children — so a long choice label is cut at the panel edge rather 

55 # than wrapping onto a second line, losing the end of the sentence that says 

56 # what the row is. The prose helpers below pass soft_wrap per call instead, 

57 # which keeps their byte-identical-to-print contract without imposing it on 

58 # the widgets. 

59 return Console(width=width, record=record, highlight=False)

62def console() -> Console: 

63 """The shared stdout console. 

64 

65 Module-level rather than injected per call site: the flows print from many 

66 places, and threading a console through every signature would churn the 

67 command modules this migration is meant to leave alone. Tests capture stdout 

68 as they already do. 

69 """ 

70 global _console 

71 if _console is None: 

72 _console = make_console() 

73 return _console

76def target(explicit: Console | None) -> Console: 

77 """Resolve the console to print on — the injected one, else the shared one. 

78 

79 Public because :mod:`calfcord.cli.tui.widgets` needs the same rule, and two 

80 modules re-deriving it invites drift: the ``or`` spelling widgets used is 

81 equivalent only because ``Console`` happens to define no ``__bool__``, which 

82 is an accident rather than a guarantee. 

83 """ 

84 return explicit if explicit is not None else console()

104def answer_text(label: str, value: str) -> Text: 

105 """Build the collapsed record: a dim tick and label, then the value, bright. 

106 

107 Built with **no base style** on purpose. A ``Text``'s base style is inherited 

108 by every appended span, so a dim base drags the value down with it — the 

109 value renders bold *and* dimmed however it was appended, and the one thing 

110 the record exists to show becomes the quietest thing on the line. Each span 

111 therefore carries its own style and nothing else. 

112 """ 

113 text = Text() 

114 text.append(f"{theme.TICK} ", style=theme.MUTED) 

115 text.append(f"{label} ", style=theme.MUTED) 

116 text.append(value, style=theme.ACCENT) 

117 return text

120def step( 

121 label: str, 

122 value: str, 

123 *, 

124 status: theme.Status = "ok", 

125 width: int = 0, 

126 console: Console | None = None, 

127) -> None: 

128 """Print one completed-step record: ``✓ label value``. 

129 

130 Shares :func:`answer`'s hierarchy — quiet glyph and label, bright value — because 

131 a step and an answer are the same shape of fact: the thing that happened matters, 

132 the name of the slot it happened in does not. It is a separate helper for two 

133 reasons. ``answer`` hard-codes a two-space gap, which is correct for a lone record 

134 collapsing out of a prompt and ragged for a block of them printed together, so 

135 ``width`` pads the label column and lets the values line up. And ``answer`` is only 

136 ever ``✓``: a step can fail, and a flow whose contract is "no green light that 

137 lies" cannot report failure with a tick. 

138 

139 ``width`` is the label column, and the caller owns it because only the caller knows 

140 the whole block; ``0`` (the default) means "no padding", for a record printed alone. 

141 A ``fail`` is styled :data:`theme.ERROR` — the theme reserves its one colour for 

142 genuine failures, and a step that did not happen is one. 

143 """ 

144 glyph = theme.STEP_GLYPHS[status] 

145 text = Text() 

146 text.append(f"{glyph} ", style=theme.ERROR if status == "fail" else theme.MUTED) 

147 text.append_text(pair_text(label, value, width=width)) 

148 target(console).print(text, soft_wrap=True)

151def pair_text(label: str, value: str, *, width: int = 0) -> Text: 

152 """Build a ``label value`` row: quiet label, bright value. No glyph. 

153 

154 Each span carries its own style and the ``Text`` carries none — a base style is 

155 inherited by every append and would drag the value dim, making the one thing the 

156 row exists to show the quietest thing on it. See :func:`answer_text`. 

157 """ 

158 text = Text() 

159 text.append(f"{label.ljust(width)} ", style=theme.MUTED) 

160 text.append(value, style=theme.ACCENT) 

161 return text

164@contextlib.contextmanager 

165def working(label: str, *, console: Console | None = None) -> Iterator[None]: 

166 """Paint a transient spinner while a slow step runs; render nothing off-TTY. 

167 

168 The arc :mod:`~calfcord.cli.tui.widgets` already performs — a transient Live torn 

169 down and replaced by a durable one-line record — applied to waiting rather than to 

170 a prompt. Reach for it only where a step is slow enough to look hung: a step that 

171 resolves in a moment is better served by its record arriving. 

172 

173 Two constraints the implementation exists to honour: 

174 

175 * **The spinner is decoration; the record is the fact.** Live is inert off a 

176 terminal (a piped run, CI, the tests), so anything painted here is *gone* there. 

177 The caller must print its :func:`step` record OUTSIDE this block, or the step 

178 vanishes from every captured log. That inertness is also what makes this safe: 

179 no frames leak into a redirected stdout. 

180 * **No hue.** Rich's default status spinner is green. In a palette whose one colour 

181 is reserved for genuine failures, a waiting spinner rendering green would be the 

182 single hued glyph in the CLI, so the style is pinned rather than inherited. 

183 """ 

184 with target(console).status( 

185 Text(label, style=theme.MUTED), spinner=theme.SPINNER, spinner_style=theme.MUTED 

186 ): 

187 yield

190def pair(label: str, value: str, *, width: int = 0, console: Console | None = None) -> None: 

191 """Print a label/value row with no outcome glyph. 

192 

193 :func:`step` minus the mark, for rows that report no outcome — the "what next" 

194 block a flow signs off with. Sharing the padded two-column shape is what lets that 

195 block read as the same object as the record board above it, rather than as unrelated 

196 prose that happens to follow. 

197 """ 

198 target(console).print(pair_text(label, value, width=width), soft_wrap=True)

210def header( 

211 title: str, 

212 *, 

213 subtitle: str = "", 

214 step: tuple[int, int] | None = None, 

215 label: str = "", 

216 console: Console | None = None, 

217) -> None: 

218 """The panel that opens an interactive command. 

219 

220 ``step`` renders as ``1/4`` in the top border, with ``label`` naming the 

221 phase — so a long wizard always says where the operator is. Single-shot 

222 commands pass neither and get a plain titled panel. 

223 """ 

224 right = "" 

225 if step is not None: 

226 current, total = step 

227 right = f"{current}/{total}" 

228 if label: 

229 right = f"{right} {theme.BULLET} {label}" 

230 

231 # A leading blank line so a header never butts against whatever preceded it 

232 # (a collapsed answer record, a note). Owned here rather than left to each 

233 # call site, which would drift. 

234 out = target(console) 

235 body = Text(subtitle, style=theme.MUTED) if subtitle else Text("") 

236 out.print() 

237 out.print( 

238 Panel( 

239 body, 

240 title=Text(title, style=theme.TITLE), 

241 title_align="left", 

242 subtitle=Text(right, style=theme.MUTED) if right else None, 

243 subtitle_align="right", 

244 border_style=theme.BORDER, 

245 box=theme.BOX, 

246 padding=(0, 1), 

247 ) 

248 )
