1"""Pure persona resolution for the bridge (C8/C9/D-7). 

2 

3After the migration the bridge no longer reads a registry for an agent's 

4``display_name``/``avatar_url``: a persona is derived entirely from the agent's 

5name — the webhook username *is* the name (C8), and the avatar is a 

6deterministic DiceBear seeded by it (C9). So :func:`persona_for` is a pure 

7function with no roster dependency, and it is handoff-correct because it is 

8keyed on the actual reply ``emitter`` (the node that really replied), not the 

9mention target. 

10""" 

11 

12from __future__ import annotations 

13 

14import zlib 

15from typing import Final 

16 

17import discord 

18 

19from calfcord.discord.avatar import dicebear_avatar_url 

20from calfcord.discord.persona import Persona 

21 

22 

23def persona_for(name: str) -> Persona: 

24 """The Discord persona for the agent ``name`` — its name as the webhook 

25 username and a deterministic DiceBear avatar seeded by that name.""" 

26 return Persona(name=name, avatar_url=dicebear_avatar_url(name)) 

27 

28 

29_ACCENTS: Final[tuple[discord.Colour, ...]] = ( 

30 discord.Colour(0x5865F2), # blurple 

31 discord.Colour(0x0AA8A0), # teal 

32 discord.Colour(0x3BA55D), # green 

33 discord.Colour(0xE67E22), # amber 

34 discord.Colour(0x9B59B6), # violet 

35 discord.Colour(0x3498DB), # blue 

36 discord.Colour(0xE91E63), # pink 

37 discord.Colour(0x1ABC9C), # mint 

38) 

39"""Hand-picked accent stripes, all legible on Discord's dark AND light themes. 

40 

41A curated palette rather than a free hue derived from the name: an arbitrary hue 

42can land muddy, or vanish against one of the two backgrounds. Collisions between 

43agents are possible and harmless — two agents sharing a stripe costs nothing, 

44whereas an illegible stripe costs the signal. 

45 

46Deliberately contains **no red**: the old renderer striped every message 

47``0xE74C3C``, successes included, so colour carried no information at all. Here 

48the stripe means *who*, and a fault is carried by the seal row. 

49""" 

50 

51 

52def accent_for(name: str) -> discord.Colour: 

53 """The trace stripe for the agent ``name`` — identity, never state. 

54 

55 Kept beside :func:`persona_for` because it is the same idea: a display 

56 attribute derived purely from the agent's name, with no roster lookup. It 

57 stays OFF :class:`Persona`, which ADR-0012 deliberately reduced to a minimal 

58 webhook identity (name + avatar); the stripe is a trace-rendering concern 

59 with one caller. 

60 

61 ``zlib.crc32``, NOT ``hash()``: Python randomises string hashing per process 

62 (``PYTHONHASHSEED``), so ``hash(name) % len`` would be perfectly stable 

63 within one run and hand the agent a *different* colour after every bridge 

64 restart. 

65 """ 

66 return _ACCENTS[zlib.crc32(name.encode("utf-8")) % len(_ACCENTS)]
