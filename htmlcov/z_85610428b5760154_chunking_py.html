1"""Chunk-splitting for Discord's 2000-character message limit. 

2 

3Agent replies and A2A audit projections can exceed Discord's hard content 

4limit; both are delivered as consecutive ≤ :data:`CHUNK_SAFE_SIZE` messages 

5split at the largest natural boundary available. Chunking is the *only* 

6delivery mechanism for long content — there is no retry that asks the agent 

7to shorten its reply. 

8 

9Consumers: :mod:`calfcord.bridge.reply_poster` (user-facing replies) and 

10:mod:`calfcord.bridge.a2a_project` (agent-to-agent audit projections). 

11 

12Process boundary: this module sits beneath its callers and depends only on 

13stdlib, so the ``bridge`` modules can import it without inducing a cycle. 

14""" 

15 

16from __future__ import annotations 

17 

18from typing import Final 

19 

20CHUNK_SAFE_SIZE: Final[int] = 1990 

21"""Max chars per chunk. Discord's hard content limit is 2000; the 10-char 

22safety buffer absorbs the occasional emoji / encoding surprise that tips a 

231999-char string over the limit.""" 

24 

25 

26def chunk_split(text: str, *, max_chars: int = CHUNK_SAFE_SIZE) -> list[str]: 

27 """Split ``text`` into pieces each ≤ ``max_chars`` for posting as 

28 consecutive Discord messages. 

29 

30 Boundary search is greedy from the largest unit down: paragraph 

31 (``"\\n\\n"``) → line (``"\\n"``) → sentence (``". "``) → word 

32 (``" "``) → hard cut. The search refuses to split earlier than 

33 ``max_chars // 2`` so we don't produce a tiny first chunk 

34 followed by a huge tail. 

35 

36 Each chunk is right-stripped of trailing whitespace. The split 

37 preserves all non-boundary characters — joining chunks back with 

38 the boundary that produced each cut reconstructs (modulo 

39 boundary whitespace) the original text. 

40 

41 Args: 

42 text: The full text to split. May be empty (returns ``[]``). 

43 max_chars: Maximum characters per chunk. Defaults to 

44 :data:`CHUNK_SAFE_SIZE` (1990) — Discord's 2000-char 

45 limit with a 10-char safety buffer. 

46 

47 Returns: 

48 A list of chunks in original order. If ``text`` already fits, 

49 returns ``[text]``. An empty string returns ``[]``. 

50 """ 

51 if not text: 

52 return [] 

53 if len(text) <= max_chars: 

54 return [text] 

55 

56 chunks: list[str] = [] 

57 remaining = text 

58 min_split = max(1, max_chars // 2) 

59 

60 while remaining: 60 ↛ 85line 60 didn't jump to line 85 because the condition on line 60 was always true

61 if len(remaining) <= max_chars: 

62 stripped = remaining.rstrip() 

63 if stripped: 

64 chunks.append(stripped) 

65 break 

66 

67 candidate = remaining[:max_chars] 

68 cut_at = -1 

69 # Prefer larger structural boundaries. 

70 for separator in ("\n\n", "\n", ". ", " "): 

71 idx = candidate.rfind(separator) 

72 if idx >= min_split: 

73 cut_at = idx + len(separator) 

74 break 

75 

76 if cut_at < 0: 

77 # No good boundary found; hard cut at max_chars. 

78 cut_at = max_chars 

79 

80 chunk = remaining[:cut_at].rstrip() 

81 if chunk: 

82 chunks.append(chunk) 

83 remaining = remaining[cut_at:] 

84 

85 return chunks
