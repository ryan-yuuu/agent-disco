1"""Chunk-splitting for Discord's message-length limits. 

2 

3Agent replies and A2A audit projections can exceed Discord's hard content 

4limit; both are delivered as consecutive ≤ :data:`CHUNK_SAFE_SIZE` messages. 

5Chunking is the *only* delivery mechanism for long content — there is no retry 

6that asks the agent to shorten its reply, so **answer content is never dropped**: 

7every character of the body survives into the chunks. Only structural framing may 

8differ — never answer text — namely trailing-boundary whitespace and some fence 

9markers (see the fence notes below). 

10 

11Whole lines are packed greedily up to the limit; a single line longer than the 

12limit is split on a sentence → word → hard boundary. Fenced code (```` ``` ````) 

13is kept whole where it fits and, where a block is too large for one message, 

14split with the fence **closed** on one chunk and **reopened** — same backticks 

15and language — on the next, so every message renders as a self-contained code 

16block rather than an orphaned half with a dangling fence. 

17 

18The same core serves both Discord's 2000-char content messages (via 

19:func:`chunk_split`) and the Components-V2 4000-char text blocks of the live 

20step trace (via :func:`calfcord.bridge.trace_rows._chunk_text`, which is a thin 

21wrapper passing its own limit): one implementation, no duplicated split logic. 

22 

23Fence detection is deliberately conservative — a line toggles fenced state only 

24when it is a *clean* marker: a run of ≥3 backticks at column 0 followed by a 

25backtick-free info string (opener), or a run of ≥ the opener's length alone 

26(closer). Consequences, all content-preserving and mostly cosmetic: 

27 

28* A line that merely *contains* backticks (prose that mentions ```` ``` ````, a 

29 ``print('```')`` inside code) is content, kept verbatim — never a dropped 

30 marker. If such content sits inside a block, Discord renders it exactly as it 

31 renders the un-chunked original; this module does not neutralise it. 

32* An indented fence, or a fence whose opener line alone is too long to wrap 

33 within the limit (a pathologically long backtick run or info string), is 

34 treated as prose — kept verbatim, never dropped, and never protected from a 

35 mid-block cut. 

36* Two adjacent blocks with the same backticks and language merge into one on 

37 render; an empty fenced block (no inner lines) is dropped. Both are rare and 

38 carry no answer content. 

39 

40Process boundary: this module sits beneath its callers and depends only on 

41stdlib, so the ``bridge`` modules can import it without inducing a cycle. 

42""" 

43 

44from __future__ import annotations 

45 

46import re 

47from itertools import groupby 

48from typing import Final, Literal, NamedTuple 

49 

50CHUNK_SAFE_SIZE: Final[int] = 1990 

51"""Max chars per chunk. Discord's hard content limit is 2000; the 10-char 

52safety buffer absorbs the occasional emoji / encoding surprise that tips a 

531999-char string over the limit.""" 

54 

55_FENCE_RUN: Final[re.Pattern[str]] = re.compile(r"`{3,}") 

56"""A run of three-or-more backticks — Discord's fence delimiter.""" 

57 

58_OPENER: Final[re.Pattern[str]] = re.compile(r"(`{3,})([^`]*)") 

59"""A clean opening fence: a leading backtick-run then a backtick-free info 

60string. Matched with ``fullmatch`` against the right-stripped line, so it never 

61matches a line that carries other content (leading text, a second run).""" 

62 

63# Separators for splitting a single over-long line, largest unit first. Newlines 

64# are already consumed by the line split, so only intra-line boundaries remain. 

65_LINE_SEPARATORS: Final[tuple[str, ...]] = (". ", " ") 

66 

67 

68class _Unit(NamedTuple): 

69 """One source line, tagged for packing. Fence markers are dropped during 

70 parsing and regenerated at render time, so a block split across chunks is 

71 re-fenced in each. ``fence``/``info`` are the block's backtick-run and info 

72 string for ``"code"`` units, empty for ``"prose"``.""" 

73 

74 kind: Literal["prose", "code"] 

75 text: str 

76 fence: str = "" 

77 info: str = "" 

78

80def chunk_split(text: str, *, max_chars: int = CHUNK_SAFE_SIZE) -> list[str]: 

81 """Split ``text`` into pieces each ≤ ``max_chars`` for posting as 

82 consecutive Discord messages, never cutting inside a fenced code block. 

83 

84 Each chunk is right-stripped of trailing whitespace; whitespace-only content 

85 yields no chunks (Discord rejects empty messages). 

86 

87 Args: 

88 text: The full text to split. Empty or whitespace-only returns ``[]``. 

89 max_chars: Maximum characters per chunk. Defaults to 

90 :data:`CHUNK_SAFE_SIZE` (1990) — Discord's 2000-char 

91 limit with a 10-char safety buffer. 

92 

93 Returns: 

94 A list of chunks in original order. If ``text`` already fits, returns 

95 ``[text]`` verbatim. 

96 """ 

97 if not text.strip(): 

98 return [] 

99 if len(text) <= max_chars: 

100 # Fits in one message — post the author's content verbatim, untouched. 

101 return [text] 

102 

103 chunks = _pack(_to_units(text, max_chars), max_chars) 

104 return [stripped for chunk in chunks if (stripped := chunk.rstrip())] 

105 

106 

107def _match_opener(line: str) -> tuple[str, str] | None: 

108 """``(backtick_run, info)`` if ``line`` is a clean opening fence, else 

109 ``None``. Requires the backticks at column 0 and an info string free of 

110 backticks, so a line carrying real content is never mistaken for a fence.""" 

111 match = _OPENER.fullmatch(line.rstrip()) 

112 return (match.group(1), match.group(2).strip()) if match else None 

113 

114 

115def _is_closer(line: str, fence_len: int) -> bool: 

116 """Whether ``line`` is a clean closing fence for a block opened with 

117 ``fence_len`` backticks: a run of at least that many backticks and nothing 

118 else. A code line that merely contains ```` ``` ```` is not a closer.""" 

119 stripped = line.rstrip() 

120 return _FENCE_RUN.fullmatch(stripped) is not None and len(stripped) >= fence_len 

121

123def _to_units(text: str, max_chars: int) -> list[_Unit]: 

124 """Tag every line as ``"prose"`` or ``"code"`` (with its fence run and info), 

125 dropping only *clean* fence-marker lines — they are regenerated at render 

126 time. An unterminated fence runs as code to the end; the renderer closes it. 

127 

128 A fence is honored only when its wrapper fits the limit (:func:`_wrappable`); 

129 a pathologically long fence run or info string — never produced by real code 

130 blocks — is left as prose so its content survives and no chunk can exceed 

131 ``max_chars``.""" 

132 units: list[_Unit] = [] 

133 fence: tuple[str, str] | None = None 

134 for line in text.split("\n"): 

135 if fence is None: 

136 opened = _match_opener(line) 

137 if opened is not None and _wrappable(opened, max_chars): 

138 fence = opened 

139 else: 

140 units.append(_Unit("prose", line)) 

141 elif _is_closer(line, len(fence[0])): 

142 fence = None 

143 else: 

144 units.append(_Unit("code", line, fence[0], fence[1])) 

145 return units 

146 

147 

148def _wrappable(opened: tuple[str, str], max_chars: int) -> bool: 

149 """Whether a fence's wrapper leaves room for at least one content char within 

150 ``max_chars``. The empty-body wrapper is ``2·len(fence) + len(info) + 2``; if 

151 that already fills the budget the block can never be split to fit, so its 

152 opener line is kept as prose instead.""" 

153 fence, info = opened 

154 return 2 * len(fence) + len(info) + 2 < max_chars

157def _render(units: list[_Unit]) -> str: 

158 """Render packed units to a chunk string, wrapping each maximal run of 

159 same-fence code units in its fence. A run confined to one chunk is fenced 

160 once; a run broken across chunks is fenced independently in each — that is 

161 the close-and-reopen that keeps every message a valid block.""" 

162 lines: list[str] = [] 

163 for (kind, fence, info), group in groupby(units, key=lambda u: (u.kind, u.fence, u.info)): 

164 texts = [unit.text for unit in group] 

165 if kind == "code": 

166 lines.append(fence + info) 

167 lines.extend(texts) 

168 lines.append(fence) 

169 else: 

170 lines.extend(texts) 

171 return "\n".join(lines)

174def _solo_len(unit: _Unit) -> int: 

175 """Rendered length of ``unit`` as its own chunk. A code unit pays its fence 

176 wrapper: opener (``fence + info``) + closer (``fence``) + two newlines.""" 

177 if unit.kind == "code": 

178 return 2 * len(unit.fence) + len(unit.info) + len(unit.text) + 2 

179 return len(unit.text)

181 

182def _append_cost(tail: tuple[str, str] | None, unit: _Unit) -> int: 

183 """Chars added by appending ``unit`` to a non-empty buffer whose trailing 

184 code run is ``tail`` (``(fence, info)``, or ``None`` when it ends in prose). 

185 Appending to an open same-fence run costs just the line; anything else opens 

186 a fresh prose line or a freshly-fenced block.""" 

187 if unit.kind == "code" and tail == (unit.fence, unit.info): 

188 return 1 + len(unit.text) 

189 return 1 + _solo_len(unit) 

190 

191 

192def _tail_of(unit: _Unit) -> tuple[str, str] | None: 

193 """The open code run a buffer ending in ``unit`` carries, or ``None``.""" 

194 return (unit.fence, unit.info) if unit.kind == "code" else None 

195

197def _split_unit(unit: _Unit, max_chars: int) -> list[_Unit]: 

198 """Split a unit whose rendered form exceeds ``max_chars`` into same-kind 

199 pieces that each fit. Prose splits on sentence/word/hard boundaries; code 

200 hard-splits (reserving room for its fence wrapper) since code has no prose 

201 boundaries — each piece stays valid because the renderer re-fences it.""" 

202 if unit.kind == "code": 

203 # _to_units only tags a line as code when its fence is wrappable, so the 

204 # wrapper leaves size >= 1 and every piece renders to <= max_chars. An 

205 # empty code line's rendered form is just the wrapper (< max_chars), so 

206 # it never reaches here. 

207 size = max_chars - _solo_len(unit._replace(text="")) 

208 return [unit._replace(text=unit.text[i : i + size]) for i in range(0, len(unit.text), size)] 

209 return [_Unit("prose", piece) for piece in _split_long_line(unit.text, max_chars)] 

210

212def _split_long_line(text: str, max_chars: int) -> list[str]: 

213 """Split one over-long line into ≤``max_chars`` pieces, preferring a 

214 sentence (``". "``) then word (``" "``) boundary no earlier than the 

215 halfway point, else a hard cut.""" 

216 pieces: list[str] = [] 

217 remaining = text 

218 min_split = max(1, max_chars // 2) 

219 while len(remaining) > max_chars: 

220 window = remaining[:max_chars] 

221 cut = -1 

222 for sep in _LINE_SEPARATORS: 

223 idx = window.rfind(sep) 

224 if idx >= min_split: 

225 cut = idx + len(sep) 

226 break 

227 if cut < 0: 

228 cut = max_chars 

229 pieces.append(remaining[:cut]) 

230 remaining = remaining[cut:] 

231 pieces.append(remaining) 

232 return pieces 

233 

234 

235def _pack(units: list[_Unit], max_chars: int) -> list[str]: 

236 """Greedily pack units into rendered chunks ≤ ``max_chars``, tracking the 

237 running rendered length so no chunk is re-rendered per unit. A unit that 

238 will not fit onto the current chunk starts a fresh one; a unit too large to 

239 fit even alone is split first, its leading pieces emitted as standalone 

240 chunks and its tail carried forward to keep packing.""" 

241 chunks: list[str] = [] 

242 buf: list[_Unit] = [] 

243 buf_len = 0 

244 tail: tuple[str, str] | None = None 

245 for unit in units: 

246 if buf: 

247 cost = _append_cost(tail, unit) 

248 if buf_len + cost <= max_chars: 

249 buf.append(unit) 

250 buf_len += cost 

251 tail = _tail_of(unit) 

252 continue 

253 chunks.append(_render(buf)) 

254 buf, buf_len, tail = [], 0, None 

255 solo = _solo_len(unit) 

256 if solo <= max_chars: 

257 buf, buf_len, tail = [unit], solo, _tail_of(unit) 

258 continue 

259 pieces = _split_unit(unit, max_chars) 

260 chunks.extend(_render([piece]) for piece in pieces[:-1]) 

261 last = pieces[-1] 

262 buf, buf_len, tail = [last], _solo_len(last), _tail_of(last) 

263 if buf: 

264 chunks.append(_render(buf)) 

265 return chunks
