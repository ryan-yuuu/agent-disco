1"""Shared agent-directory inspection and ``.md`` write helpers for the CLI. 

2 

3Both ``disco init`` (which *reports* the agents an install would load) and 

4``disco agent tools`` (which *picks* one to edit) need the same answer to 

5"which ``.md`` files are live agents?". Factoring :func:`detect_agents` here 

6keeps that one definition from drifting between the two callers — a mismatch 

7would let ``init`` report an agent the editor can't open, or vice versa. 

8 

9This module also owns the agent-file *write* and *identity* helpers that both 

10``disco init`` (first-run setup) and ``disco agent create`` build on: 

11slugifying a typed name into a valid stem, deriving a default display name and 

12body, the create/update :func:`write_agent` path, and the tools-checkbox 

13builder :func:`pick_tools`. They live here rather than in ``init`` so the two 

14commands share one implementation and can't drift. 

15 

16The skip rules in :func:`detect_agents` mirror the loader's 

17(:func:`calfcord.agents.loader.load_agents_dir`): dot-prefixed files and 

18``*.template.md`` reference templates are not live agents, so the names returned 

19here match exactly what ``calfkit-agent`` would run. 

20 

21:func:`pick_tools` defers its ``TOOL_REGISTRY`` import into the function body so 

22the rest of the module (e.g. :func:`detect_agents`) stays light — importing the 

23registry eagerly composes the tool surface (importing the vendored 

24``calfkit-tools`` nodes). 

25"""

55# The starter agent's name and the *exact* description the installer seeds it 

56# with. The prune-pristine check in :func:`write_agent` keys off this string: 

57# an ``assistant.md`` still carrying it is an untouched seed (safe to remove 

58# when the operator names a different agent); any other description means the 

59# operator customized it and it must be preserved. 

60STARTER_AGENT_NAME = "assistant" 

61DEFAULT_DESCRIPTION = "General-purpose AI teammate — answers questions and helps with tasks." 

62 

63# Tools that grant code-execution or filesystem-write reach into the 

64# ``calfkit-tools`` host. Selecting any of them drives the one-line security 

65# caution, because anyone who can !mention the agent can then drive them. 

66# ``terminal`` and ``execute_code`` run arbitrary code on the host (see 

67# docs/adr/0005); ``write_file``/``patch`` mutate the shared workspace. 

68_DANGEROUS_TOOLS = frozenset( 

69 {"terminal", "process", "execute_code", "write_file", "patch"} 

70)

72# Permitted characters for an agent-name *stem*. The on-disk identifier must 

73# satisfy ``AgentDefinition.agent_id`` (``[a-z0-9_-]{1,32}``); we slugify toward 

74# that here so a friendly typed name ("My Helper") becomes a valid stem 

75# ("my_helper") rather than failing validation at write time. 

76_STEM_INVALID = re.compile(r"[^a-z0-9_-]+") 

77 

78# The create row's value, and why this exact spelling is safe. It shares a 

79# namespace with agent names (both are ``Choice.value``s in one select, and 

80# ``ListState`` requires those to be unique), so it must be a string no agent 

81# can ever be called — a collision does not merely mis-render, it RAISES, so the 

82# picker never opens and no agent can be started at all. 

83# 

84# The leading separator is what buys that, and it is worth being exact about why, 

85# because the obvious argument is wrong: ``AGENT_ID_PATTERN`` does NOT keep this 

86# namespace clean. It governs the WRITE path (``slug_stem`` rewrites a typed name 

87# toward it), while :func:`detect_agents` — which supplies every other row — 

88# globs raw ``.md`` stems and validates nothing. An earlier version of this 

89# constant was ``+create`` and reasoned that ``+`` was outside the pattern; a 

90# hand-placed ``+create.md`` duly collided and crashed the picker. A stem is a 

91# single path component, so it can never contain ``/`` — that holds no matter 

92# what ``detect_agents`` filters, which is the property this needs. 

93# 

94# It is also not a valid agent id, so were it ever to leak through to 

95# ``roster.agent_start`` it would fail that command's name-shape check loudly 

96# rather than start something unexpected. That is a backstop, not the guarantee. 

97CREATE_SENTINEL = "/create" 

98 

99# ``+`` reads as "new"; the ellipsis is the long-standing convention for a row 

100# that OPENS something rather than acting immediately. Together they do the work 

101# a colour would do elsewhere — deliberately, since :mod:`calfcord.cli.tui.theme` 

102# is monochrome and every unselected row is already dim, leaving no styling 

103# headroom to mark this row as different. 

104_CREATE_LABEL = "+ Create a new agent…" 

105 

106 

107# The synthetic checkbox row that represents MCP discover mode (``mcp: true``). 

108# Deliberately colon-separated, not ``mcp/``-prefixed, so it never collides with a 

109# named ``mcp/<server>`` grant row when the split partitions the selection. 

110MCP_DISCOVER_ROW = "mcp:discover"

127def pick_agent( 

128 prompter: Prompter, 

129 *, 

130 agents_dir: Path, 

131 message: str, 

132 create_fn: Callable[[], str | None], 

133) -> str | None: 

134 """Prompt for one of the DEFINED agents — or for a brand-new one. 

135 

136 The "which agent?" pick-list. Returns the chosen agent's name; ``None`` means 

137 "nothing to start", and whatever produced it has already said why, so callers 

138 map it onto exit 1 without adding a message of their own. 

139 

140 The list always ends with a ``+ Create a new agent…`` row, because "none of 

141 these" was the one answer it could not take: an operator whose roster held 

142 nothing they wanted had to quit and run another command. Choosing it returns 

143 whatever ``create_fn`` produces — a new agent's name, or ``None`` if the 

144 create failed (in which case ``create_fn`` owns reporting it). 

145 

146 ``create_fn`` is injected rather than imported because 

147 :mod:`calfcord.cli.agent_create` imports *this* module, so reaching the other 

148 way would cycle; it also keeps this module's one job — asking the question — 

149 separate from the flow that answers it. It is REQUIRED, not defaulted: the 

150 only caller always creates, so a ``None`` default would be a branch no run 

151 ever reaches, and it would let the next caller acquire a create-less picker 

152 without ever deciding to. 

153 

154 Placement is deliberate. :class:`~calfcord.cli.tui.state.SelectState` opens on 

155 row 0, so a create row at the top would make launching a wizard the 

156 enter-through default of a command whose usual intent is to start an agent 

157 that already exists. Last costs nothing in reach: ``ListState`` navigation 

158 wraps, so one ``↑`` lands on it however long the roster is. 

159 

160 An empty roster opens the picker rather than refusing it, and the reasoning 

161 here used to run the other way: a choice-less list is unanswerable — no key 

162 means "none of these" — so it stranded the operator with only Ctrl-C, and 

163 naming the command that makes an agent was the honest reply. Offering to 

164 create inverts that: the list is no longer choice-less, it holds exactly one 

165 honest answer, and offering it beats printing a command to go and type. 

166 

167 This lists what is **defined** on disk, not what is **running**. That suits 

168 ``start``; it would be wrong for ``stop``/``restart``, whose real question is 

169 "which of the running agents?" — answerable only with a broker probe, and a 

170 defined-agent list there would invite picking one that is already stopped. 

171 

172 ``agent edit`` / ``agent tools`` keep their own near-identical pick-lists 

173 rather than calling this, and that is deliberate: they return a ``Path``, they 

174 have a name-given branch this has no equivalent of, and their empty-roster 

175 dead end is honest — there is nothing to edit, and offering to create inside 

176 "which agent do you want to edit?" answers a different question than the one 

177 asked. The part that could actually drift between them, ``detect_agents``, is 

178 already shared. 

179 """ 

180 agents = detect_agents(agents_dir) 

181 if not agents: 

182 # The dead-end this replaced still carried one fact worth keeping: WHICH 

183 # directory was searched is how an operator recognizes a wrong 

184 # ``$CALFCORD_HOME``. As dim detail above the picker it survives without 

185 # the dead end it used to arrive with. 

186 render.note(f"no agents in {agents_dir}") 

187 

188 choices = [Choice(a, a) for a in agents] 

189 choices.append(Choice(CREATE_SENTINEL, _CREATE_LABEL)) 

190 

191 chosen = prompter.select(message, choices) 

192 return create_fn() if chosen == CREATE_SENTINEL else chosen

195def detect_agents(agents_dir: Path) -> list[str]: 

196 """Return the agent names (``.md`` stems) ``agents_dir`` would load, sorted. 

197 

198 Returns an empty list when ``agents_dir`` is not an existing directory, so 

199 callers can treat "no dir" and "empty dir" identically — both mean "no 

200 agents to act on". Dotfiles and ``*.template.md`` templates are skipped to 

201 match the loader; the result is sorted for deterministic prompts/output. 

202 """ 

203 if not agents_dir.is_dir(): 

204 return [] 

205 return sorted( 

206 p.stem 

207 for p in agents_dir.glob("*.md") 

208 if not p.name.startswith(".") and not p.name.endswith(".template.md") 

209 )

212def slug_stem(raw: str) -> str: 

213 """Coerce a typed agent name into a safe ``.md`` filename stem. 

214 

215 The frontmatter ``name`` (and thus the filename) must match 

216 ``[a-z0-9_-]{1,32}``; an operator typing "My Helper" should not hit a 

217 validation error, so we lowercase, turn runs of disallowed characters into 

218 single underscores, and trim the result. A name that slugifies to nothing 

219 (e.g. all punctuation) falls back to the starter name rather than producing 

220 an empty, invalid stem — keeping the wizard moving instead of aborting. 

221 """ 

222 slug = _STEM_INVALID.sub("_", raw.strip().lower()).strip("_-") 

223 slug = slug[:32].strip("_-") 

224 result = slug or STARTER_AGENT_NAME 

225 # Postcondition: the stem is a valid agent id, so callers can write it as a 

226 # filename without a second validation. The fallback guarantees non-empty, 

227 # and slugification guarantees the charset/length, so this can only fire on 

228 # a bug in those rules — assert rather than silently emit an invalid stem. 

229 assert AGENT_ID_PATTERN.fullmatch(result), f"slugified stem {result!r} is not a valid agent id" 

230 return result

250def agent_body(name: str) -> str: 

251 """Render the generic system-prompt body for a brand-new agent. 

252 

253 A minimal, generic prompt so the agent answers sensibly from the first boot 

254 without further editing. Addresses the agent by a human-friendly rendering of 

255 its slug ``name`` (underscores/dashes title-cased, "my_helper" → "My Helper") 

256 so the greeting reads naturally. Kept separate from the frontmatter so the 

257 create path can serialize identity fields through ``frontmatter.dumps`` (which 

258 YAML-quotes free-text values safely) rather than string interpolation. 

259 """ 

260 human = name.replace("_", " ").replace("-", " ").title() 

261 return ( 

262 f"You are {human}, a helpful AI teammate in this Discord workspace. Answer\n" 

263 "questions and help with tasks clearly and concisely. If you don't know something,\n" 

264 "say so rather than guessing.\n" 

265 "\n" 

266 "You talk to people through Discord, so you can use Discord-flavored markdown in\n" 

267 "your replies. Tables do not render.\n" 

268 )

271def atomic_write(path: Path, payload: str) -> None: 

272 """Write ``payload`` to ``path`` via a same-dir tmp file + atomic rename. 

273 

274 A partial agent file would make the next ``calfkit-agent`` boot fail to 

275 load the directory, so the create path must never leave a half-written file 

276 behind on error. ``path.parent`` is created first because a fresh install's 

277 ``agents/`` directory may not exist yet (unlike :mod:`md_writer`'s in-place 

278 rewrite, which can assume the file — and so the dir — already exists). 

279 """ 

280 path.parent.mkdir(parents=True, exist_ok=True) 

281 fd, tmp_name = tempfile.mkstemp(dir=path.parent, prefix=f".{path.name}.", suffix=".tmp") 

282 tmp_path = Path(tmp_name) 

283 try: 

284 with os.fdopen(fd, "w", encoding="utf-8") as tmp: 

285 tmp.write(payload) 

286 tmp.flush() 

287 os.fsync(tmp.fileno()) 

288 os.replace(tmp_path, path) 

289 except BaseException: 

290 tmp_path.unlink(missing_ok=True) 

291 raise

294def is_pristine_seed(agents_dir: Path) -> bool: 

295 """True when ``agents_dir/assistant.md`` is the untouched seeded starter. 

296 

297 "Untouched" is detected by the seed's two stable identity markers: its 

298 ``agent_id`` is still ``assistant`` and its description is still the exact 

299 seed default. If the operator customized the description, both halves no 

300 longer hold and the file must be preserved. A missing or malformed 

301 ``assistant.md`` is treated as "not a pristine seed" (nothing safe to 

302 prune), so the parse is guarded — a broken file is never deleted on a guess. 

303 """ 

304 seed = agents_dir / f"{STARTER_AGENT_NAME}.md" 

305 if not seed.is_file(): 

306 return False 

307 try: 

308 parsed = parse_agent_md(seed) 

309 except (ValueError, OSError): 

310 return False 

311 return parsed.agent_id == STARTER_AGENT_NAME and parsed.description == DEFAULT_DESCRIPTION

320def _ensure_memory_tools(tools: list[str] | None, *, memory: bool) -> list[str] | None: 

321 """Return ``tools`` with memory's required filesystem tools present when needed. 

322 

323 Memory-enabled agents manage their notepad with ``read_file`` / ``write_file``. 

324 An omitted ``tools:`` (``None``) already discovers every builtin and is fine. 

325 An explicit list that dropped either tool would otherwise pass create and 

326 fail later at agent build — so top them up here and leave ordering stable 

327 (required tools appended only when missing). 

328 """ 

329 if not memory or tools is None: 

330 return tools 

331 ensured = list(tools) 

332 for required in _MEMORY_REQUIRED_TOOLS: 

333 if required not in ensured: 

334 ensured.append(required) 

335 return ensured

338def write_agent( 

339 agents_dir: Path, 

340 *, 

341 name: str, 

342 description: str, 

343 provider: str, 

344 model: str, 

345 tools: list[str] | None, 

346 mcp: bool | list[str] | tuple[str, ...] = True, 

347 memory: bool = True, 

348 prune_seed: bool = False, 

349) -> Path: 

350 """Create or update ``agents_dir/<name>.md`` for the wizard's agent. 

351 

352 Two paths, both validate-before-write so a bad value never lands on disk: 

353 

354 * **Target exists** — update the agent in place, preserving its body: 

355 rewrite ``description``/``provider``/``model`` via 

356 :func:`md_writer._update_fields`, then the split tool grants via 

357 :func:`md_writer.update_tool_grants`. Both are validated-atomic, so a bad 

358 value leaves the file untouched. The existing ``memory`` setting is left 

359 alone — re-running create against an on-disk agent must not silently flip 

360 it — but if that existing value is on, the same ``read_file`` / 

361 ``write_file`` top-up applied on create runs against the new tools list 

362 so an update cannot leave a memory-enabled agent factory-rejected. 

363 * **Target missing** — build the frontmatter as a mapping and serialize it 

364 with :func:`frontmatter.dumps` (NOT string interpolation), which 

365 YAML-quotes free-text values so a description like ``"Calendar: book 

366 meetings"`` or one carrying quotes/``#``/leading punctuation can't corrupt 

367 the file. New agents default to ``memory: true`` (written explicitly) so 

368 ``disco agent create`` / ``disco init`` teammates start with the 

369 persistent notepad on; the schema default for an *omitted* field stays 

370 ``false`` so hand-authored and pre-existing agents are unchanged. When 

371 memory is on and ``tools`` is an explicit list missing ``read_file`` / 

372 ``write_file``, those tools are added before validation so the wizard 

373 cannot produce an agent the factory would reject. The synthetic 

374 :class:`AgentDefinition` is built *first* (mirroring 

375 :func:`md_writer._update_fields`), so an invalid value raises before any 

376 disk write. After the atomic write, when ``prune_seed`` is set and the 

377 operator named a *different* agent (``name != "assistant"``) on an install 

378 still carrying the *pristine* seeded ``assistant.md``, that seed is deleted 

379 so they end with one clean agent. ``init`` opts in for its first-run setup; 

380 ``agent create`` leaves it off so a second agent never removes the starter. 

381 A *customized* ``assistant.md`` (or naming the agent ``assistant`` itself) 

382 is never deleted. 

383 

384 Raises: 

385 ValueError: a field value fails :class:`AgentDefinition` validation 

386 (create path) or the existing ``.md``/new value is invalid (update 

387 path). No partial file is written. 

388 OSError: a filesystem error during the atomic write. No partial file is 

389 written. 

390 """ 

391 target = agents_dir / f"{name}.md" 

392 

393 if target.exists(): 

394 # Honor the on-disk memory bit (do not apply the create-time default), 

395 # but still satisfy the factory's fs-tool requirement when it is on. 

396 existing_memory = parse_agent_md(target).memory 

397 tools = _ensure_memory_tools(tools, memory=existing_memory) 

398 md_writer._update_fields(target, {"description": description, "provider": provider, "model": model}) 

399 md_writer.update_tool_grants(target, tools=tools, mcp=mcp) 

400 return target 

401 

402 tools = _ensure_memory_tools(tools, memory=memory) 

403 body = agent_body(name) 

404 metadata = { 

405 "name": name, 

406 "description": description, 

407 "provider": provider, 

408 "model": model, 

409 "memory": memory, 

410 } 

411 if tools is not None: 

412 metadata["tools"] = list(tools) 

413 # Shared with the update path (md_writer.update_tool_grants) so the two 

414 # cannot diverge on how a tri-state serializes. 

415 md_writer.apply_mcp_metadata(metadata, mcp) 

416 # Validate the full definition in memory FIRST (mirrors 

417 # md_writer._update_fields): a bad free-text value raises here, before any 

418 # bytes touch disk, so the create path can never leave an unloadable file. 

419 AgentDefinition(**{**metadata, "system_prompt": body, "source_path": target}) 

420 

421 payload = frontmatter.dumps(frontmatter.Post(body, **metadata)) 

422 if not payload.endswith("\n"): 422 ↛ 424line 422 didn't jump to line 424 because the condition on line 422 was always true

423 payload += "\n" 

424 atomic_write(target, payload) 

425 

426 # Prune the pristine starter only when the caller opted in (``init``'s 

427 # first-run "one clean agent" goal) and a *different* agent was created; 

428 # naming the agent ``assistant`` would have hit the update path above. 

429 # ``agent create`` leaves ``prune_seed`` False so adding a second agent 

430 # never deletes the operator's starter. 

431 if prune_seed and name != STARTER_AGENT_NAME and is_pristine_seed(agents_dir): 

432 (agents_dir / f"{STARTER_AGENT_NAME}.md").unlink(missing_ok=True) 

433 logger.info("pruned pristine seed assistant.md after creating %s", target) 

434 

435 return target

438def pick_tools( 

439 prompter: Prompter, 

440 name: str, 

441 *, 

442 mcp_servers_fn: Callable[[], list[str]] | None = None, 

443 live_tools_fn: Callable[[], dict[str, list[str]]] | None = None, 

444) -> ToolGrantSelection: 

445 """Prompt for the agent's tools and return split builtin/MCP grants. 

446 

447 Every default builtin (tools-host registry plus bridge Discord reads) is 

448 offered pre-checked so the default is the same "all builtins" set a 

449 frontmatter that omits ``tools:`` discovers at runtime, and the MCP discover 

450 row is likewise pre-checked so a wizard-created agent matches a hand-authored 

451 one (which defaults to ``mcp: true``). The *named* MCP rows (``mcp/<server>`` 

452 from mcp.json plus live-discovered per-tool rows) start UNCHECKED — deselect 

453 discover to opt out or to pick named servers instead. Row building is shared 

454 with the ``agent tools`` editor (:func:`calfcord.cli.agent_tools._build_choices`) 

455 so the two surfaces can't drift. If a write/shell tool ends up selected we 

456 print the security caution, because anyone who can !mention the agent can 

457 then drive it. 

458 """ 

459 from calfcord.cli.agent_tools import ( 

460 _build_choices, 

461 _default_live_tools, 

462 _default_mcp_servers, 

463 ) 

464 from calfcord.tools import default_builtin_tool_names 

465 

466 default_builtins = set(default_builtin_tool_names()) 

467 choices = _build_choices( 

468 # Builtins AND MCP-discover pre-checked, so a wizard-created agent's default 

469 # matches a hand-authored one (omitted ``tools:`` + ``mcp: true``); named MCP 

470 # rows start unchecked. Deselect discover to opt out or to pick named servers. 

471 default_builtins | {MCP_DISCOVER_ROW}, 

472 mcp_servers=(mcp_servers_fn or _default_mcp_servers)(), 

473 live_tools=(live_tools_fn or _default_live_tools)(), 

474 ) 

475 

476 selected = prompter.checkbox( 

477 f"Tools for {name}", 

478 choices, 

479 # The title asks; the instruction explains. This guidance used to be a 

480 # parenthetical inside the title, which turned the question into a 

481 # paragraph — and left ``instruction`` unused and looking like dead 

482 # speculation. It deliberately does NOT restate the key mechanics: the 

483 # hint in the panel's border already says space/enter for every list. 

484 instruction="All selected — deselect any you don't want.", 

485 ) 

486 

487 if _DANGEROUS_TOOLS.intersection(selected): 

488 print( 

489 "note: these tools include code execution + file write access in the " 

490 "calfkit-tools launch dir, drivable by anyone who can !mention this agent " 

491 "— keep the bot off public Discord (docs/security.md §3.4)." 

492 ) 

493 

494 return _split_tool_selection(selected, default_builtins)

497def _split_tool_selection(selected: list[str], builtin_names: set[str]) -> ToolGrantSelection: 

498 """Convert checkbox UI values into canonical frontmatter fields. 

499 

500 The MCP surface is tri-state: the synthetic :data:`MCP_DISCOVER_ROW` maps to 

501 ``mcp=True`` (discover every live server) and, being exclusive, subsumes any 

502 named ``mcp/<server>`` rows also ticked; otherwise the named rows form the 

503 grant list, and an empty selection is ``[]`` (opt out). 

504 """ 

505 builtin_tokens = [ 

506 token for token in selected if not token.startswith("mcp/") and token != MCP_DISCOVER_ROW 

507 ] 

508 selected_known_builtins = {token for token in builtin_tokens if token in builtin_names} 

509 has_unknown_builtin = any(token not in builtin_names for token in builtin_tokens) 

510 

511 tools = None if not has_unknown_builtin and selected_known_builtins == builtin_names else builtin_tokens
