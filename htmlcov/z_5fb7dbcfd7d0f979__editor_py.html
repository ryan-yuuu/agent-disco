1"""Resolving which editor to launch for long-form text, and how. 

2 

3Split out of :mod:`calfcord.cli.agent_edit` because getting this right needs 

4more care than a one-liner suggests, and the reasoning deserves a home. Each 

5rule below was checked against a real implementation, not against the folklore 

6around the ``$EDITOR`` convention. 

7 

8The old one-liner — ``os.environ.get("EDITOR") or "vi"`` — had two real defects 

9and one unkind default: 

10 

11* It ignored ``VISUAL``, which every implementation checked honours **first** 

12 (click, gh, gemini-cli, Codex, aider). Setting only ``VISUAL`` is common, and 

13 we silently ignored it and dropped the operator into vi. 

14* It never told a GUI editor to wait, so ``EDITOR=code`` returned instantly and 

15 the operator's edit was **silently discarded** (see :data:`_WAIT_FLAGS`). 

16* It fell back to bare ``vi``. Being dropped unannounced into a modal editor is 

17 the single most-reported way a CLI strands a newcomer, and this wizard's whole 

18 metric is time-to-first-reply. 

19""" 

20 

21from __future__ import annotations 

22 

23import os 

24import shlex 

25import shutil 

26from pathlib import Path 

27 

28# GUI editors fork and return immediately unless told to wait for the window to 

29# close. We would then read the temp file back before a single keystroke landed, 

30# conclude "unchanged", and throw the operator's work away without a word — a 

31# silent data loss, the worst kind. gemini-cli injects these for the same reason. 

32# Keyed by binary name; the value is that editor's own spelling of "block". 

33_WAIT_FLAGS = { 

34 "code": "--wait", 

35 "code-insiders": "--wait", 

36 "codium": "--wait", 

37 "cursor": "--wait", 

38 "windsurf": "--wait", 

39 "zed": "--wait", 

40 "atom": "--wait", 

41 "gedit": "--wait", 

42 "subl": "-w", 

43 "sublime_text": "-w", 

44 "mate": "-w", 

45} 

46 

47# Probed in order when neither VISUAL nor EDITOR is set. Deliberately a probe 

48# rather than a constant: the field does not agree on a default (gh and jj ship 

49# nano, click prefers vim, aider and gemini-cli use vi), so any single hardcoded 

50# name is a guess that fails on some boxes. prompt_toolkit hardcodes /usr/bin 

51# paths and silently breaks under Homebrew and Nix — the failure mode to avoid. 

52# 

53# ``sensible-editor`` first because on Debian it IS this decision, made by the 

54# distro and the user's own `select-editor` choice. Then the friendly editors 

55# before the modal one: this list only runs when the operator has expressed no 

56# preference at all, and someone with no $EDITOR is unlikely to want vi. 

57_FALLBACKS = ("sensible-editor", "nano", "vim", "vi") 

58 

59# Last resort when even the probe finds nothing. Codex raises here instead; we 

60# would rather try and fail visibly than refuse to open an editor at all. 

61_LAST_RESORT = "vi" 

62 

63 

64def resolve() -> list[str]: 

65 """Return the editor to launch, as an argv prefix (no filename). 

66 

67 A prefix rather than a string because the value may carry arguments 

68 (``EDITOR="emacs -nw"``), and because :func:`argv` must append the filename 

69 **last** — a path before a flag is not honoured. 

70 """ 

71 for key in ("VISUAL", "EDITOR"): 

72 value = os.environ.get(key) 

73 if value and value.strip(): 

74 # shlex so a command line ("code --wait") is honoured rather than 

75 # treated as one impossible binary name. 

76 return _ensure_wait(shlex.split(value)) 

77 

78 for candidate in _FALLBACKS: 

79 if shutil.which(candidate): 

80 return [candidate] 

81 return [_LAST_RESORT] 

82

84def _ensure_wait(command: list[str]) -> list[str]: 

85 """Append the wait flag if this is a GUI editor that needs one. 

86 

87 Matches on the *basename*, so ``/usr/local/bin/code`` is recognised. Never 

88 duplicates a flag the operator already set, and never adds one to a terminal 

89 editor — vim would reject ``--wait`` as an unknown option. 

90 """ 

91 if not command: 91 ↛ 92line 91 didn't jump to line 92 because the condition on line 91 was never true

92 return command 

93 flag = _WAIT_FLAGS.get(Path(command[0]).name) 

94 if flag is None or flag in command: 

95 return command 

96 return [*command, flag]

98 

99def describe(command: list[str]) -> str: 

100 """The editor's name, for telling the operator what is about to open. 

101 

102 Just the binary — arguments are noise in a sentence, and a full path is 

103 worse than a name. Naming it is the cheap half of gh's 

104 ``[(e) to launch nano, enter to skip]``: the failure being prevented is an 

105 operator staring at an unfamiliar full-screen editor with no idea what it is. 

106 """ 

107 return Path(command[0]).name if command else _LAST_RESORT 

108 

109 

110def argv(command: list[str], path: Path) -> list[str]: 

111 """The full argv: the editor prefix, then the file. The file goes last.""" 

112 return [*command, str(path)]
