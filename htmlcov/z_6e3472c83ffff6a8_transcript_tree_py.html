1"""The persisted transcript's tree renderer — pure, no Discord, no Kafka, no 

2state (spec §5.1). 

3 

4**Full transcript** — :func:`_render_tree_blocks` projects a turn's 

5``message_history`` slice into the Claude-Code-style ``● tool(args)`` / 

6``⎿ result`` blocks; its block COUNT gates whether the reply poster persists a 

7transcript row (for tool-call replay). It operates on ``Sequence[ModelMessage]`` 

8because it renders persisted deltas, and its output is byte-for-byte stable so 

9stored transcripts keep rendering the same. 

10 

11This module once also held the live step renderer. That surface is now row 

12values (:mod:`calfcord.bridge.trace_rows`) posted by 

13:class:`~calfcord.bridge.trace.StepTraceRenderer`: rows are markdown rather than 

14fenced code, so the two shared nothing once the split landed.

38# --- Full transcript tree renderer (the persisted step transcript) ---------- 

39# Renders the turn's steps as a Claude-Code-style trace: model text as prose, 

40# each tool call as ``● tool(args)`` with its result nested under ``⎿``. This is 

41# the FULL trace — NO per-part truncation. One string per visual block (a prose 

42# block, or a call+return pair); the block COUNT gates whether the reply poster 

43# persists the turn's transcript row (for tool-call replay), so a tool call and 

44# its result count as ONE step. 

45 

46_TREE_CALL_MARKER: Final[str] = "●" 

47_TREE_RETURN_MARKER: Final[str] = "⎿" 

48 

49_TRIPLE_BACKTICK_RUN: Final[re.Pattern[str]] = re.compile(r"`{3,}") 

50

55def _fence_safe(content: str) -> str: 

56 """Neutralize runs of 3+ backticks so embedded fences can't break out. 

57 

58 Discord closes a ``` code block at the next run of three-or-more 

59 backticks regardless of the opening fence's length, so a triple-backtick 

60 inside tool output would otherwise terminate the block early and spill the 

61 remainder as raw markdown. Weaving a zero-width space between the backticks 

62 of any 3+ run leaves the text visually identical while ensuring no raw 

63 ``` survives to close the fence. Single/double backticks are left untouched 

64 — they render literally inside a block. (The only cost: a steps.md copied 

65 from output that itself contained ``` carries invisible zero-width spaces.) 

66 """ 

67 return _TRIPLE_BACKTICK_RUN.sub(lambda m: "\u200b".join("`" * len(m.group())), content) 

68 

69 

70def _fenced(content: str) -> str: 

71 """Wrap ``content`` in a code fence, neutralizing any inner ``` first.""" 

72 return f"```\n{_fence_safe(content)}\n```"

75def _format_call_args(args: dict[str, Any]) -> str: 

76 """Render tool-call args as keyword form: ``k=<json-value>, …``. 

77 

78 ``args`` is the already-coerced argument dict from :func:`_tool_call_args`, 

79 which guarantees a dict (an unparseable arg is coerced to ``{}`` and renders 

80 as ``name()``). A flat object renders as ``city="Tokyo", n=5`` — each value 

81 JSON-encoded, so strings keep their quotes and booleans read as 

82 ``true``/``false``; nested values use compact separators to keep the line 

83 tight. Byte fidelity is preserved: JSON already escapes real newlines, so 

84 the signature stays one line and inner spacing in string values survives. 

85 """ 

86 if not args: 

87 return "" 

88 pairs: list[str] = [] 

89 for key, value in args.items(): 

90 try: 

91 rendered_value = json.dumps(value, separators=(",", ":"), default=str) 

92 except (TypeError, ValueError): 

93 rendered_value = str(value) 

94 pairs.append(f"{key}={rendered_value}") 

95 return ", ".join(pairs)

98def _tool_call_args(call: ToolCallPart) -> dict[str, Any]: 

99 """Best-effort argument dict for a transcript ``ToolCallPart`` (never raises). 

100 

101 The tree renderer projects persisted ``message_history``, so it coerces a 

102 tool call's raw args here: a non-object or unparseable arg becomes ``{}`` 

103 (rendered as ``name()``) rather than raising and losing the WHOLE stored 

104 transcript over one malformed call. ``args_as_dict`` is pydantic-ai's 

105 canonical accessor — it returns ``{}`` for empty args and asserts on a 

106 bare list/scalar — so well-formed object args render byte-for-byte as before. 

107 """ 

108 try: 

109 args = call.args_as_dict() 

110 except Exception: 

111 # A bare list/scalar makes args_as_dict assert; malformed JSON makes it 

112 # raise. Either way the args aren't an object → render as ``name()``. 

113 return {} 

114 return args if isinstance(args, dict) else {}

120def _render_text_part(part: TextPart) -> str | None: 

121 """Render a ``TextPart`` into a transcript block, or ``None`` to skip. 

122 

123 Whitespace-only content is skipped — empty preambles are common 

124 when the model emits a tool call with no narrative. 

125 """ 

126 text = part.content.strip() 

127 if not text: 

128 return None 

129 return text

132def _tool_tree_block(call: ToolCallPart, ret: ToolReturnPart | None) -> str: 

133 """Render a tool call and its (optional) result as one fenced tree block. 

134 

135 ``● tool(args)`` on the first line; when a matching return is present, its 

136 result is nested under ``⎿`` with continuation lines aligned beneath the 

137 first result character. Args use the keyword form WITHOUT whitespace 

138 collapsing — this is the full view, so byte fidelity is preserved (real 

139 newlines are already JSON-escaped, so the signature stays one line). No 

140 truncation: the only bound is the overall message cap enforced upstream. 

141 """ 

142 sig = f"{call.tool_name}({_format_call_args(_tool_call_args(call))})" 

143 lines = [f"{_TREE_CALL_MARKER} {sig}"] 

144 if ret is not None: 

145 first, *rest = ret.model_response_str().split("\n") 

146 lines.append(f" {_TREE_RETURN_MARKER} {first}") 

147 lines.extend(f" {line}" for line in rest) 

148 return _fenced("\n".join(lines)) 

149 

150 

151def _return_tree_block(ret: ToolReturnPart) -> str: 

152 """Render an orphan tool return (no call with its id in the slice) standalone. 

153 

154 Practically unreachable — a tool call and its return live in the same 

155 agent run, after the history cursor, so they're sliced together. Rendered 

156 defensively so an orphan return is never silently dropped, which would also 

157 skew the step count. 

158 """ 

159 first, *rest = ret.model_response_str().split("\n") 

160 lines = [f"{_TREE_RETURN_MARKER} {first}"] 

161 lines.extend(f" {line}" for line in rest) 

162 return _fenced("\n".join(lines))

165def _render_tree_blocks(messages: Sequence[ModelMessage]) -> list[str]: 

166 """Project the turn's ``message_history`` slice into full tree blocks. 

167 

168 The source of the reply poster's step COUNT (``len(...)``), which gates 

169 whether a transcript row is persisted for tool-call replay. Walks the delta 

170 in order, emitting one string per visual block: 

171 

172 * a model ``TextPart`` → a prose block (whitespace-only skipped); 

173 * a ``ToolCallPart`` → ``● tool(args)`` with its matched ``ToolReturnPart`` 

174 (looked up by ``tool_call_id``) nested under ``⎿`` — a call and its 

175 result are ONE block, so the step count credits a tool use once. 

176 

177 Skips non-rendered parts (``ThinkingPart``, ``FilePart``, 

178 ``BuiltinTool*Part``, ``UserPromptPart`` / ``SystemPromptPart``, 

179 ``RetryPromptPart``). 

180 

181 Pairing is purely by id and independent of message order: a return is 

182 folded into its call iff a call with that id exists anywhere in the slice; 

183 a return whose call is absent renders standalone (so nothing is dropped, 

184 and the orphan path can't double-render a return that arrives before its 

185 call). Output order follows message order. Duplicate ``tool_call_id``s 

186 don't occur in well-formed pydantic-ai history; on a collision the last 

187 return for an id wins. 

188 

189 Caller wraps this in a try/except — ``model_response_str`` can raise on 

190 malformed payloads. 

191 """ 

192 # Two index passes (order-independent): which ids have a call in the 

193 # slice, and the return for each id. A return is then an orphan iff its id 

194 # has no call here — decided without relying on walk order. 

195 call_ids: set[str] = set() 

196 returns_by_id: dict[str, ToolReturnPart] = {} 

197 for msg in messages: 

198 if isinstance(msg, ModelResponse): 

199 for part in msg.parts: 

200 if isinstance(part, ToolCallPart): 

201 call_ids.add(part.tool_call_id) 

202 elif isinstance(msg, ModelRequest): 202 ↛ 197line 202 didn't jump to line 197 because the condition on line 202 was always true

203 for part in msg.parts: 

204 if isinstance(part, ToolReturnPart): 

205 returns_by_id[part.tool_call_id] = part 

206 

207 out: list[str] = [] 

208 for msg in messages: 

209 if isinstance(msg, ModelResponse): 

210 for part in msg.parts: 

211 if isinstance(part, TextPart): 

212 rendered = _render_text_part(part) 

213 if rendered is not None: 

214 out.append(rendered) 

215 elif isinstance(part, ToolCallPart): 215 ↛ 210line 215 didn't jump to line 210 because the condition on line 215 was always true

216 out.append(_tool_tree_block(part, returns_by_id.get(part.tool_call_id))) 

217 elif isinstance(msg, ModelRequest): 217 ↛ 208line 217 didn't jump to line 208 because the condition on line 217 was always true

218 for part in msg.parts: 

219 if isinstance(part, ToolReturnPart) and part.tool_call_id not in call_ids: 

220 out.append(_return_tree_block(part)) 

221 return out
