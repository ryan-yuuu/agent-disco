1"""Render A2A activity (consults + handoffs) into the unified Discord audit channel. 

2 

3The stateful :class:`~calfcord.bridge.a2a_dispatch.A2ADispatcher` pulls native 

4``message_agent`` consults and ``HandoffEvent``s off a run's stream and emits the 

5:class:`~calfcord.bridge.a2a_dispatch.A2AProjection` dataclasses; this projector 

6turns each into Discord posts in the unified A2A channel (re-homed from the old 

7``private_chat`` tool, spec §6.2). It is the ``A2AProjectorLike`` collaborator the 

8bridge's :class:`~calfcord.bridge.mention_handler.MentionHandler` drives. 

9 

10Anchoring (round-3 M3): one thread per **``correlation_id``** (one human turn's A2A 

11activity), created lazily on the first projection for that turn — its message is the 

12thread's starter. Every later request/reply/reject/handoff/fault for the same 

13``correlation_id`` posts into that thread. Peer identity comes from the projection 

14dataclasses (already resolved by the dispatcher from the request's ``args["name"]``, 

15the one source stable across success and rejection); personas come from the pure 

16:func:`~calfcord.bridge.persona_resolve.persona_for` (no roster). 

17 

18Best-effort audit: the bridge is no longer the A2A *transport* (the consult already 

19happened inside the agent runtime and its reply is in-hand on the stream), so a 

20failed Discord render is logged and swallowed — it never faults the human turn. 

21"""

68"""Soft cap on the topic-tail portion (after ``caller→peer: ``)."""

70"""Substituted when the seed content is empty, avoiding a bare-trailing-space name."""

73def _build_thread_name(caller: str, peer: str, content: str) -> str: 

74 """Produce a thread name like ``'conan→scribe: please summarize the doc'``. 

75 

76 Control characters are normalized to spaces and runs collapsed; the topic tail 

77 is truncated to :data:`_THREAD_NAME_CONTENT_MAX` and the whole name hard-capped 

78 at :data:`_THREAD_NAME_MAX_TOTAL` (Discord's limit). The ``→`` (U+2192) 

79 separator is char-counted (not byte-counted) by Discord, so it is cheap. 

80 """ 

81 cleaned = " ".join("".join(c if c.isprintable() else " " for c in content).split()) 

82 if not cleaned: 

83 cleaned = _THREAD_NAME_EMPTY_PLACEHOLDER 

84 name = f"{caller}→{peer}: {cleaned[:_THREAD_NAME_CONTENT_MAX]}" 

85 return name[:_THREAD_NAME_MAX_TOTAL]

95def _request_preview(message: str) -> str: 

96 """A short, single-line glimpse of a consult's prompt for its trace row. 

97 

98 Truncation is on the raw text; the row's own ``_plain`` then flattens and 

99 markdown-escapes it, so this need only decide *how much* to show. May be empty 

100 (a blank prompt) — the row renders as inline regardless, so a blank ask shows 

101 a bare marker rather than a link or the audit-gap. 

102 """ 

103 trimmed = message.strip() 

104 if len(trimmed) <= _REQUEST_PREVIEW_MAX: 

105 return trimmed 

106 return trimmed[:_REQUEST_PREVIEW_MAX].rstrip() + "…"

109class A2AProjector: 

110 """Renders :class:`A2AProjection`s into the unified A2A audit channel. 

111 

112 One instance per bridge; its ``correlation_id → thread_id`` map is the only 

113 state, mirroring one thread per human turn's A2A activity. 

114 """ 

115 

116 def __init__( 

117 self, 

118 resolver: A2AChannelResolver, 

119 personas: DiscordPersonaSender, 

120 steps: StepTraceRenderer, 

121 ) -> None: 

122 self._resolver = resolver 

123 self._personas = personas 

124 self._threads: dict[str, int] = {} 

125 # The audit channel's OWN step renderer — a DIFFERENT instance from the 

126 # one rendering the human's thread (ADR-0026). Two surfaces, two entry 

127 # maps: both are keyed by correlation_id and both see the same one for a 

128 # turn, so sharing an instance would collide them onto one trace. 

129 self._steps = steps 

130 self._channel_id: int | None = None 

131 self._degraded = False 

132 

133 async def project(self, projection: A2AProjection) -> str | None: 

134 """Render one projection; return the audit thread's jump URL, or ``None``. 

135 

136 The URL is this render's **receipt** — returned only when the post actually 

137 reached Discord — so the bridge can cross-link the exchange it just wrote. 

138 ``None`` means the render failed and was swallowed (best-effort: a Discord 

139 failure must never fault the human turn), leaving nothing to link to. 

140 

141 Deriving the link from the thread map instead would answer a subtly 

142 different question — "does a thread exist for this turn?" — which diverges 

143 the moment a turn's SECOND consult fails: the first consult's thread is 

144 still mapped, so the caller would confidently link a thread that never 

145 received this exchange. A receipt cannot drift from what was written. 

146 """ 

147 thread_id = await self._guarded(self._dispatch(projection), type(projection).__name__) 

148 if thread_id is None: 

149 return None 

150 return f"https://discord.com/channels/{self._resolver.guild_id}/{thread_id}" 

151 

152 async def _guarded(self, render: Awaitable[int], kind: str) -> int | None: 

153 """Run one best-effort render: swallow a Discord failure (it must never 

154 fault the human turn — this runs inside the mention handler's stream-drain 

155 loop), note the audit gap, and report the thread it reached or ``None``. 

156 

157 Both render entry points funnel through here so the gap latch arms *and 

158 re-arms* identically on either — an asymmetry would silently lose the loud 

159 line for the next real outage. 

160 """ 

161 try: 

162 thread_id = await render 

163 except Exception: 

164 self._note_gap(kind) 

165 return None 

166 self._degraded = False # this outage is over; a LATER one earns its own loud line 

167 return thread_id

169 def _note_gap(self, kind: str) -> None: 

170 """Log an audit gap: loudly the first time, quietly while it persists. 

171 

172 The projection is best-effort, so a broken audit channel produces no 

173 user-visible error and no fault — only this line. It is therefore an ERROR 

174 naming the remedy, not a bare WARN. Repeats drop to DEBUG because the 

175 failure is almost always systemic (a missing permission fails identically 

176 on every consult), and re-logging a traceback per projection buries the 

177 one line that matters under its own noise. 

178 """ 

179 if self._degraded: 

180 # exc_info stays on: the latch is a bare bool, so a DIFFERENT failure 

181 # arriving during an outage lands here too — without the traceback it 

182 # would vanish with no diagnostic at any level. 

183 logger.debug("A2A projection still failing (audit gap); continuing kind=%s", kind, exc_info=True) 

184 return 

185 self._degraded = True 

186 logger.error( 

187 "A2A projection failed (audit gap); continuing kind=%s. %s", 

188 kind, 

189 _AUDIT_GAP_REMEDY, 

190 exc_info=True, 

191 )

205 async def project_step(self, step: StepEvent) -> None: 

206 """Fold one CONSULTED agent's step into its trace in that turn's thread 

207 (ADR-0026). The drain hands us every step whose emitter is not the turn's 

208 acting agent; the acting agent's own steps go to the human's thread. 

209 

210 ``acting_agent=step.emitter`` renders every step under the agent that 

211 actually emitted it — in an audit view a tool call belongs to the agent 

212 that made it, and a nested consult's peer is its own participant. It goes 

213 through the SAME renderer the human's thread uses, so 23 tool calls are 

214 one edited-in-place message rather than 23 (ADR-0017), rendered in the 

215 same row grammar (ADR-0024). 

216 

217 A step never CREATES the thread: a thread is named ``caller→peer`` from 

218 the consult, which a step cannot supply, and a step cannot precede its own 

219 consult. So no thread means the request's render failed (best-effort) — 

220 drop, matching the audit gap already logged there. 

221 """ 

222 thread_id = self._threads.get(step.correlation_id) 

223 if thread_id is None: 

224 logger.debug( 

225 "A2A: no thread for correlation=%s; dropping consulted agent's step emitter=%s kind=%s", 

226 step.correlation_id, 

227 step.emitter, 

228 step.kind, 

229 ) 

230 return 

231 channel_id = self._channel_id 

232 if channel_id is None: # pragma: no cover - a thread implies a channel 

233 return 

234 await self._steps.on_step( 

235 step, 

236 Destination(channel_id=channel_id, thread_id=thread_id), 

237 acting_agent=step.emitter, 

238 )

240 async def project_consult(self, request: A2ARequest) -> None: 

241 """Announce a NESTED consult as a resolving row in the CALLER's trace 

242 (ADR-0027): ``◐ consulting <peer>`` under the caller's persona, with a 

243 glimpse of the ask folded on. Its peer's work then renders below in the 

244 same thread, so the peer no longer appears unannounced. 

245 

246 This REPLACES the standalone prompt message for a nested consult — the 

247 drain does not also ``project`` the request — so the row is the single, 

248 resolving signal rather than a bare ``[caller] <prompt>`` line. The row is 

249 ``inline`` (its exchange is in this thread, nothing to link), so it never 

250 renders the top-level audit-gap marker even when the ask is blank. 

251 

252 Thread lookup mirrors :meth:`project_step`: a nested consult cannot precede 

253 the top-level one that named the thread, so a missing thread means that 

254 render failed (best-effort) — drop, rather than anchor an unnameable one. 

255 """ 

256 thread_id = self._threads.get(request.correlation_id) 

257 if thread_id is None: 

258 logger.debug( 

259 "A2A: no thread for correlation=%s; dropping nested consult row caller=%s peer=%s", 

260 request.correlation_id, 

261 request.caller, 

262 request.peer, 

263 ) 

264 return 

265 channel_id = self._channel_id 

266 if channel_id is None: # pragma: no cover - a thread implies a channel 

267 return 

268 await self._steps.on_consult( 

269 request.tool_call_id, 

270 request.peer, 

271 None, # the exchange is inline in THIS thread — nothing to link to 

272 Destination(channel_id=channel_id, thread_id=thread_id), 

273 correlation_id=request.correlation_id, 

274 persona_name=request.caller, 

275 inline=True, 

276 request_preview=_request_preview(request.message), 

277 )

281 

282 The state/note mapping is shared with the human-thread row via 

283 :func:`~calfcord.bridge.a2a_dispatch.consult_outcome`. A never-seen key or 

284 correlation is silently ignored by the renderer (its row may have been 

285 dropped for want of a thread), matching the best-effort contract.

288 await self._steps.on_consult_result( 

289 projection.tool_call_id, 

290 state=state, 

291 note=note, 

292 correlation_id=projection.correlation_id, 

293 )

295 async def seal(self, correlation_id: str, *, faulted: bool) -> None: 

296 """Close the consulted agents' trace with the run's outcome (ADR-0025). 

297 

298 Driven by the same stream terminal that seals the human's thread. Without 

299 it ``finish`` below would seal every consulted trace defensively as 

300 ``interrupted`` — wrong for a clean run, and it would bury the case this 

301 whole surface exists for: an agent that made 23 tool calls and then 

302 faulted must read "run failed after 23 tools", not simply stop. 

303 """ 

304 await self._steps.seal(correlation_id, faulted=faulted) 

305 

306 async def finish(self, correlation_id: str) -> None: 

307 """Retire the turn's projector state. 

308 

309 MUST run after the terminal reply: a faulted run synthesises its 

310 dangling-consult notes during delivery, and ``_emit`` needs the turn's 

311 thread still mapped — evicting first would create a SECOND thread and 

312 orphan each note in it. 

313 

314 The eviction is unconditional (a ``finally``) because ``_threads`` was 

315 previously never evicted at all. That was tolerable at ~100 bytes per 

316 turn; it is not now the projector owns a renderer, because an unretired 

317 entry strands a live asyncio writer task per turn. 

318 """ 

319 try: 

320 await self._steps.finish(correlation_id) 

321 finally: 

322 self._threads.pop(correlation_id, None)

324 async def _dispatch(self, projection: A2AProjection) -> int: 

325 """Render one projection and return the thread it landed in.""" 

326 if isinstance(projection, A2ARequest): 

327 return await self._emit( 

328 projection.correlation_id, 

329 persona_for(projection.caller), 

330 projection.message, 

331 thread_name=_build_thread_name(projection.caller, projection.peer, projection.message), 

332 ) 

333 elif isinstance(projection, A2AReply): 

334 return await self._emit( 

335 projection.correlation_id, 

336 persona_for(projection.peer), 

337 projection.text, 

338 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

339 ) 

340 elif isinstance(projection, A2AReject): 

341 # A rejected consult (peer offline / cycle / self) is a system note, 

342 # not a peer post — the peer never spoke (D-2). 

343 return await self._emit( 

344 projection.correlation_id, 

345 _SYSTEM_PERSONA, 

346 f"⚠️ consult to {projection.peer} was rejected: {projection.text}", 

347 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

348 ) 

349 elif isinstance(projection, A2AFailed): 349 ↛ 363line 349 didn't jump to line 363 because the condition on line 349 was always true

350 # A consult that reached the peer but faulted — a system note distinct 

351 # from a refused dispatch (A2AReject); the peer engaged but errored. 

352 return await self._emit( 

353 projection.correlation_id, 

354 _SYSTEM_PERSONA, 

355 f"💥 consult to {projection.peer} failed: {projection.text}", 

356 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

357 ) 

358 else: 

359 # Exhaustiveness guard: a 4th A2AProjection variant added without a 

360 # branch here is a mypy error, not a silent no-op render. (Handoffs 

361 # are NOT projected here — they render inline in the main step stream 

362 # via the step-trace renderer; the dispatcher no longer emits them.) 

363 assert_never(projection)

370 async def _emit(self, correlation_id: str, persona: Persona, content: str, *, thread_name: str) -> int: 

371 """Post ``content`` under ``persona`` into ``correlation_id``'s thread, and 

372 return that thread's id. 

373 

374 Creates the thread lazily on the first projection for the turn — the first 

375 content chunk becomes the thread's anchor/starter message — then posts any 

376 remaining chunks into the thread. Content over Discord's 2000-char limit is 

377 split (:func:`chunk_split`); empty content uses :data:`_EMPTY_PLACEHOLDER`. 

378 """ 

379 channel_id = await self._channel() 

380 chunks = chunk_split(content) or [_EMPTY_PLACEHOLDER] 

381 thread_id = self._threads.get(correlation_id) 

382 if thread_id is None: 

383 sent = await self._personas.send(persona, channel_id=channel_id, content=chunks[0]) 

384 thread_id = await self._resolver.create_anchored_thread(channel_id, sent.id, name=thread_name) 

385 self._threads[correlation_id] = thread_id 

386 rest = chunks[1:] 

387 else: 

388 rest = chunks 

389 for chunk in rest: 

390 await self._personas.send(persona, channel_id=channel_id, content=chunk, thread_id=thread_id) 

391 return thread_id
