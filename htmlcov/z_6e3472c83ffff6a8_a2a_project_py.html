1"""Render A2A activity (consults + handoffs) into the unified Discord audit channel. 

2 

3The stateful :class:`~calfcord.bridge.a2a_dispatch.A2ADispatcher` pulls native 

4``message_agent`` consults and ``HandoffEvent``s off a run's stream and emits the 

5:class:`~calfcord.bridge.a2a_dispatch.A2AProjection` dataclasses; this projector 

6turns each into Discord posts in the unified A2A channel (re-homed from the old 

7``private_chat`` tool, spec §6.2). It is the ``A2AProjectorLike`` collaborator the 

8bridge's :class:`~calfcord.bridge.mention_handler.MentionHandler` drives. 

9 

10Anchoring (round-3 M3): one thread per **``correlation_id``** (one human turn's A2A 

11activity), created lazily on the first projection for that turn — its message is the 

12thread's starter. Every later request/reply/reject/handoff/fault for the same 

13``correlation_id`` posts into that thread. Peer identity comes from the projection 

14dataclasses (already resolved by the dispatcher from the request's ``args["name"]``, 

15the one source stable across success and rejection); personas come from the pure 

16:func:`~calfcord.bridge.persona_resolve.persona_for` (no roster). 

17 

18Best-effort audit: the bridge is no longer the A2A *transport* (the consult already 

19happened inside the agent runtime and its reply is in-hand on the stream), so a 

20failed Discord render is logged and swallowed — it never faults the human turn. 

21"""

61# Thread-name shaping (re-homed from the old private_chat tool). 

62_THREAD_NAME_MAX_TOTAL = 100 

63"""Discord's hard cap on thread names; exceeding it 400s the create.""" 

64_THREAD_NAME_CONTENT_MAX = 40 

65"""Soft cap on the topic-tail portion (after ``caller→peer: ``).""" 

66_THREAD_NAME_EMPTY_PLACEHOLDER = "<empty>" 

67"""Substituted when the seed content is empty, avoiding a bare-trailing-space name."""

70def _build_thread_name(caller: str, peer: str, content: str) -> str: 

71 """Produce a thread name like ``'conan→scribe: please summarize the doc'``. 

72 

73 Control characters are normalized to spaces and runs collapsed; the topic tail 

74 is truncated to :data:`_THREAD_NAME_CONTENT_MAX` and the whole name hard-capped 

75 at :data:`_THREAD_NAME_MAX_TOTAL` (Discord's limit). The ``→`` (U+2192) 

76 separator is char-counted (not byte-counted) by Discord, so it is cheap. 

77 """ 

78 cleaned = " ".join("".join(c if c.isprintable() else " " for c in content).split()) 

79 if not cleaned: 

80 cleaned = _THREAD_NAME_EMPTY_PLACEHOLDER 

81 name = f"{caller}→{peer}: {cleaned[:_THREAD_NAME_CONTENT_MAX]}" 

82 return name[:_THREAD_NAME_MAX_TOTAL]

85class A2AProjector: 

86 """Renders :class:`A2AProjection`s into the unified A2A audit channel. 

87 

88 One instance per bridge; its ``correlation_id → thread_id`` map is the only 

89 state, mirroring one thread per human turn's A2A activity. 

90 """ 

91 

92 def __init__(self, resolver: A2AChannelResolver, personas: DiscordPersonaSender) -> None: 

93 self._resolver = resolver 

94 self._personas = personas 

95 self._threads: dict[str, int] = {} 

96 self._channel_id: int | None = None 

97 self._degraded = False 

98 

99 async def project(self, projection: A2AProjection) -> str | None: 

100 """Render one projection; return the audit thread's jump URL, or ``None``. 

101 

102 The URL is this render's **receipt** — returned only when the post actually 

103 reached Discord — so the bridge can cross-link the exchange it just wrote. 

104 ``None`` means the render failed and was swallowed (best-effort: a Discord 

105 failure must never fault the human turn), leaving nothing to link to. 

106 

107 Deriving the link from the thread map instead would answer a subtly 

108 different question — "does a thread exist for this turn?" — which diverges 

109 the moment a turn's SECOND consult fails: the first consult's thread is 

110 still mapped, so the caller would confidently link a thread that never 

111 received this exchange. A receipt cannot drift from what was written. 

112 """ 

113 thread_id = await self._guarded(self._dispatch(projection), type(projection).__name__) 

114 if thread_id is None: 

115 return None 

116 return f"https://discord.com/channels/{self._resolver.guild_id}/{thread_id}" 

117 

118 async def _guarded(self, render: Awaitable[int], kind: str) -> int | None: 

119 """Run one best-effort render: swallow a Discord failure (it must never 

120 fault the human turn — this runs inside the mention handler's stream-drain 

121 loop), note the audit gap, and report the thread it reached or ``None``. 

122 

123 Both render entry points funnel through here so the gap latch arms *and 

124 re-arms* identically on either — an asymmetry would silently lose the loud 

125 line for the next real outage. 

126 """ 

127 try: 

128 thread_id = await render 

129 except Exception: 

130 self._note_gap(kind) 

131 return None 

132 self._degraded = False # this outage is over; a LATER one earns its own loud line 

133 return thread_id 

134 

135 def _note_gap(self, kind: str) -> None: 

136 """Log an audit gap: loudly the first time, quietly while it persists. 

137 

138 The projection is best-effort, so a broken audit channel produces no 

139 user-visible error and no fault — only this line. It is therefore an ERROR 

140 naming the remedy, not a bare WARN. Repeats drop to DEBUG because the 

141 failure is almost always systemic (a missing permission fails identically 

142 on every consult), and re-logging a traceback per projection buries the 

143 one line that matters under its own noise. 

144 """ 

145 if self._degraded: 

146 # exc_info stays on: the latch is a bare bool, so a DIFFERENT failure 

147 # arriving during an outage lands here too — without the traceback it 

148 # would vanish with no diagnostic at any level. 

149 logger.debug("A2A projection still failing (audit gap); continuing kind=%s", kind, exc_info=True) 

150 return 

151 self._degraded = True 

152 logger.error( 

153 "A2A projection failed (audit gap); continuing kind=%s. %s", 

154 kind, 

155 _AUDIT_GAP_REMEDY, 

156 exc_info=True, 

157 )

159 async def project_fault(self, call: A2ACall) -> None: 

160 """Note a consult that never got a reply because the peer faulted (D-2).""" 

161 await self._guarded( 

162 self._emit( 

163 call.correlation_id, 

164 _SYSTEM_PERSONA, 

165 f"⚠️ {call.peer} did not reply — the consult faulted before a response.", 

166 thread_name=_build_thread_name(call.caller, call.peer, call.message), 

167 ), 

168 "A2AFault", 

169 )

171 async def _dispatch(self, projection: A2AProjection) -> int: 

172 """Render one projection and return the thread it landed in.""" 

173 if isinstance(projection, A2ARequest): 

174 return await self._emit( 

175 projection.correlation_id, 

176 persona_for(projection.caller), 

177 projection.message, 

178 thread_name=_build_thread_name(projection.caller, projection.peer, projection.message), 

179 ) 

180 elif isinstance(projection, A2AReply): 

181 return await self._emit( 

182 projection.correlation_id, 

183 persona_for(projection.peer), 

184 projection.text, 

185 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

186 ) 

187 elif isinstance(projection, A2AReject): 

188 # A rejected consult (peer offline / cycle / self) is a system note, 

189 # not a peer post — the peer never spoke (D-2). 

190 return await self._emit( 

191 projection.correlation_id, 

192 _SYSTEM_PERSONA, 

193 f"⚠️ consult to {projection.peer} was rejected: {projection.text}", 

194 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

195 ) 

196 elif isinstance(projection, A2AFailed): 196 ↛ 210line 196 didn't jump to line 210 because the condition on line 196 was always true

197 # A consult that reached the peer but faulted — a system note distinct 

198 # from a refused dispatch (A2AReject); the peer engaged but errored. 

199 return await self._emit( 

200 projection.correlation_id, 

201 _SYSTEM_PERSONA, 

202 f"💥 consult to {projection.peer} failed: {projection.text}", 

203 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

204 ) 

205 else: 

206 # Exhaustiveness guard: a 4th A2AProjection variant added without a 

207 # branch here is a mypy error, not a silent no-op render. (Handoffs 

208 # are NOT projected here — they render inline in the main step stream 

209 # via the step-trace renderer; the dispatcher no longer emits them.) 

210 assert_never(projection)

212 async def _channel(self) -> int: 

213 if self._channel_id is None: 

214 self._channel_id = await self._resolver.resolve_unified_channel() 

215 return self._channel_id

217 async def _emit(self, correlation_id: str, persona: Persona, content: str, *, thread_name: str) -> int: 

218 """Post ``content`` under ``persona`` into ``correlation_id``'s thread, and 

219 return that thread's id. 

220 

221 Creates the thread lazily on the first projection for the turn — the first 

222 content chunk becomes the thread's anchor/starter message — then posts any 

223 remaining chunks into the thread. Content over Discord's 2000-char limit is 

224 split (:func:`chunk_split`); empty content uses :data:`_EMPTY_PLACEHOLDER`. 

225 """ 

226 channel_id = await self._channel() 

227 chunks = chunk_split(content) or [_EMPTY_PLACEHOLDER] 

228 thread_id = self._threads.get(correlation_id) 

229 if thread_id is None: 

230 sent = await self._personas.send(persona, channel_id=channel_id, content=chunks[0]) 

231 thread_id = await self._resolver.create_anchored_thread(channel_id, sent.id, name=thread_name) 

232 self._threads[correlation_id] = thread_id 

233 rest = chunks[1:] 

234 else: 

235 rest = chunks 

236 for chunk in rest: 

237 await self._personas.send(persona, channel_id=channel_id, content=chunk, thread_id=thread_id) 

238 return thread_id
