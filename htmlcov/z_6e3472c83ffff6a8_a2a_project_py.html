1"""Render A2A activity (consults + handoffs) into the unified Discord audit channel. 

2 

3The stateful :class:`~calfcord.bridge.a2a_dispatch.A2ADispatcher` pulls native 

4``message_agent`` consults and ``HandoffEvent``s off a run's stream and emits the 

5:class:`~calfcord.bridge.a2a_dispatch.A2AProjection` dataclasses; this projector 

6turns each into Discord posts in the unified A2A channel (re-homed from the old 

7``private_chat`` tool, spec §6.2). It is the ``A2AProjectorLike`` collaborator the 

8bridge's :class:`~calfcord.bridge.mention_handler.MentionHandler` drives. 

9 

10Anchoring (round-3 M3): one thread per **``correlation_id``** (one human turn's A2A 

11activity), created lazily on the first projection for that turn — its message is the 

12thread's starter. Every later request/reply/reject/handoff/fault for the same 

13``correlation_id`` posts into that thread. Peer identity comes from the projection 

14dataclasses (already resolved by the dispatcher from the request's ``args["name"]``, 

15the one source stable across success and rejection); personas come from the pure 

16:func:`~calfcord.bridge.persona_resolve.persona_for` (no roster). 

17 

18Best-effort audit: the bridge is no longer the A2A *transport* (the consult already 

19happened inside the agent runtime and its reply is in-hand on the stream), so a 

20failed Discord render is logged and swallowed — it never faults the human turn. 

21"""

73def _build_thread_name(root_agent: str, content: str) -> str: 

74 """Produce a turn-level thread name like ``'marketing · plan launch'``. 

75 

76 Control characters are normalized to spaces and runs collapsed; the topic tail 

77 The leading routing token is removed when it names ``root_agent``; thread 

78 identity describes the human turn, never whichever parallel peer happened to 

79 be projected first. 

80 """ 

81 cleaned = " ".join("".join(c if c.isprintable() else " " for c in content).split()) 

82 cleaned = re.sub(rf"^!{re.escape(root_agent)}(?:\s+|$)", "", cleaned, flags=re.IGNORECASE).strip() 

83 cleaned = cleaned or _THREAD_NAME_EMPTY_PLACEHOLDER 

84 name = f"{root_agent} · {cleaned}" 

85 return name[:_THREAD_NAME_MAX_TOTAL]

198 async def project(self, projection: A2AProjection) -> str | None: 

199 """Render one projection; return the audit thread's jump URL, or ``None``. 

200 

201 The URL is this render's **receipt** — returned only when the post actually 

202 reached Discord — so the bridge can cross-link the exchange it just wrote. 

203 ``None`` means the render failed and was swallowed (best-effort: a Discord 

204 failure must never fault the human turn), leaving nothing to link to. 

205 

206 Deriving the link from the thread map instead would answer a subtly 

207 different question — "does a thread exist for this turn?" — which diverges 

208 the moment a turn's SECOND consult fails: the first consult's thread is 

209 still mapped, so the caller would confidently link a thread that never 

210 received this exchange. A receipt cannot drift from what was written. 

211 """ 

212 thread_id = await self._guarded(self._dispatch(projection), type(projection).__name__) 

213 if thread_id is None or thread_id <= 0: 

214 return None 

215 return f"https://discord.com/channels/{self._resolver.guild_id}/{thread_id}"

217 async def _guarded(self, render: Awaitable[int], kind: str) -> int | None: 

218 """Run one best-effort render: swallow a Discord failure (it must never 

219 fault the human turn — this runs inside the mention handler's stream-drain 

220 loop), note the audit gap, and report the thread it reached or ``None``. 

221 

222 Both render entry points funnel through here so the gap latch arms *and 

223 re-arms* identically on either — an asymmetry would silently lose the loud 

224 line for the next real outage. 

225 """ 

226 try: 

227 thread_id = await render 

228 except Exception: 

229 self._note_gap(kind) 

230 return None 

231 self._degraded = False # this outage is over; a LATER one earns its own loud line 

232 return thread_id

234 def _note_gap(self, kind: str) -> None: 

235 """Log an audit gap: loudly the first time, quietly while it persists. 

236 

237 The projection is best-effort, so a broken audit channel produces no 

238 user-visible error and no fault — only this line. It is therefore an ERROR 

239 naming the remedy, not a bare WARN. Repeats drop to DEBUG because the 

240 failure is almost always systemic (a missing permission fails identically 

241 on every consult), and re-logging a traceback per projection buries the 

242 one line that matters under its own noise. 

243 """ 

244 if self._degraded: 

245 # exc_info stays on: the latch is a bare bool, so a DIFFERENT failure 

246 # arriving during an outage lands here too — without the traceback it 

247 # would vanish with no diagnostic at any level. 

248 logger.debug("A2A projection still failing (audit gap); continuing kind=%s", kind, exc_info=True) 

249 return 

250 self._degraded = True 

251 logger.error( 

252 "A2A projection failed (audit gap); continuing kind=%s. %s", 

253 kind, 

254 _AUDIT_GAP_REMEDY, 

255 exc_info=True, 

256 )

258 async def project_fault(self, call: A2ACall) -> None: 

259 """Resolve the original card when a consult never produced a result.""" 

260 turn = self._turns.get(call.correlation_id) 

261 if turn is None or call.tool_call_id not in turn.branches: 

262 # Nested consults have inline rows, not cards. A request whose card 

263 # failed to open has already logged the real audit gap. Neither case 

264 # is a second Discord outage and must not poison the gap latch. 

265 return 

266 await self._guarded(self._resolve_branch(call.correlation_id, call.tool_call_id, "interrupted"), "A2AFault")

268 async def project_step(self, step: StepEvent) -> None: 

269 """Fold one CONSULTED agent's step into its trace in that turn's thread 

270 (ADR-0026). The drain hands us every step whose emitter is not the turn's 

271 acting agent; the acting agent's own steps go to the human's thread. 

272 

273 ``acting_agent=step.emitter`` renders every step under the agent that 

274 actually emitted it — in an audit view a tool call belongs to the agent 

275 that made it, and a nested consult's peer is its own participant. It goes 

276 through the SAME renderer the human's thread uses, so 23 tool calls are 

277 one edited-in-place message rather than 23 (ADR-0017), rendered in the 

278 same row grammar (ADR-0024). 

279 

280 A step never CREATES the thread: the first consult's route card anchors 

281 the turn-level thread, and a step cannot precede its own consult. So no 

282 thread means the request's render failed (best-effort) — 

283 drop, matching the audit gap already logged there. 

284 """ 

285 thread_id = self._threads.get(step.correlation_id) 

286 if thread_id is None: 

287 logger.debug( 

288 "A2A: no thread for correlation=%s; dropping consulted agent's step emitter=%s kind=%s", 

289 step.correlation_id, 

290 step.emitter, 

291 step.kind, 

292 ) 

293 return 

294 channel_id = self._channel_id 

295 if channel_id is None: # pragma: no cover - a thread implies a channel 

296 return 

297 await self._steps.on_step( 

298 step, 

299 Destination(channel_id=channel_id, thread_id=thread_id), 

300 acting_agent=step.emitter, 

301 )

303 async def project_consult(self, request: A2ARequest) -> None: 

304 """Announce a NESTED consult as a resolving row in the CALLER's trace 

305 (ADR-0027): ``◐ consulting <peer>`` under the caller's persona, with a 

306 glimpse of the ask folded on. Its peer's work then renders below in the 

307 same thread, so the peer no longer appears unannounced. 

308 

309 This REPLACES the standalone prompt message for a nested consult — the 

310 drain does not also ``project`` the request — so the row is the single, 

311 resolving signal rather than a bare ``[caller] <prompt>`` line. The row is 

312 ``inline`` (its exchange is in this thread, nothing to link), so it never 

313 renders the top-level audit-gap marker even when the ask is blank. 

314 

315 Thread lookup mirrors :meth:`project_step`: a nested consult cannot precede 

316 the top-level one that named the thread, so a missing thread means that 

317 render failed (best-effort) — drop, rather than anchor an unnameable one. 

318 """ 

319 thread_id = self._threads.get(request.correlation_id) 

320 if thread_id is None: 

321 logger.debug( 

322 "A2A: no thread for correlation=%s; dropping nested consult row caller=%s peer=%s", 

323 request.correlation_id, 

324 request.caller, 

325 request.peer, 

326 ) 

327 return 

328 channel_id = self._channel_id 

329 if channel_id is None: # pragma: no cover - a thread implies a channel 

330 return 

331 await self._steps.on_consult( 

332 request.tool_call_id, 

333 request.peer, 

334 None, # the exchange is inline in THIS thread — nothing to link to 

335 Destination(channel_id=channel_id, thread_id=thread_id), 

336 correlation_id=request.correlation_id, 

337 persona_name=request.caller, 

338 inline=True, 

339 request_preview=_request_preview(request.message), 

340 )

358 async def seal(self, correlation_id: str, *, faulted: bool) -> None: 

359 """Close the consulted agents' trace with the run's outcome (ADR-0025). 

360 

361 Driven by the same stream terminal that seals the human's thread. Without 

362 it ``finish`` below would seal every consulted trace defensively as 

363 ``interrupted`` — wrong for a clean run, and it would bury the case this 

364 whole surface exists for: an agent that made 23 tool calls and then 

365 faulted must read "run failed after 23 tools", not simply stop. 

366 """

369 async def finish(self, correlation_id: str) -> None: 

370 """Retire the turn's projector state. 

371 

372 MUST run after terminal delivery: a faulted run synthesises dangling 

373 consult outcomes there, and their cards need the turn's thread mapping. 

374 

375 The eviction is unconditional (a ``finally``) because ``_threads`` was 

376 previously never evicted at all. That was tolerable at ~100 bytes per 

377 turn; it is not now the projector owns a renderer, because an unretired 

378 entry strands a live asyncio writer task per turn. 

379 """ 

380 turn = self._turns.get(correlation_id) 

381 if turn is not None: 381 ↛ 389line 381 didn't jump to line 389 because the condition on line 381 was always true

382 for tool_call_id, branch in list(turn.branches.items()): 

383 if branch.state != "pending": 

384 continue 

385 await self._guarded( 

386 self._resolve_branch(correlation_id, tool_call_id, "interrupted"), 

387 "A2AInterruptedCard", 

388 ) 

389 try: 

390 await self._steps.finish(correlation_id) 

391 finally: 

392 self._threads.pop(correlation_id, None) 

393 self._turns.pop(correlation_id, None)

395 async def _dispatch(self, projection: A2AProjection) -> int: 

396 """Render one projection and return the thread it landed in.""" 

397 if isinstance(projection, A2ARequest): 

398 return await self._open_branch(projection) 

399 elif isinstance(projection, A2AReply): 

400 turn = self._turns.get(projection.correlation_id) 

401 branch = turn.branches.get(projection.tool_call_id) if turn is not None else None 

402 if branch is not None and turn is not None: 

403 # Card edits are cosmetic: a failed edit must not suppress the peer's 

404 # substantive response, so it has its own best-effort boundary. 

405 try: 

406 thread_id = await self._resolve_branch( 

407 projection.correlation_id, 

408 projection.tool_call_id, 

409 "replied", 

410 record_state=False, 

411 ) 

412 except Exception: 

413 self._note_gap("A2AReplyCard") 

414 thread_id = turn.thread_id or 0 

415 # If thread creation failed after the card reached the parent, 

416 # preserve the answer beside the card but return no jump receipt. 

417 await self._post_response(projection, turn.thread_id, branch=branch) 

418 return thread_id 

419 

420 # Nested requests are represented by an inline trace row, not a 

421 # standalone card. Their reply still needs its explicit return route. 

422 thread_id = self._threads.get(projection.correlation_id, 0) 

423 if thread_id <= 0: 423 ↛ 425line 423 didn't jump to line 425 because the condition on line 423 was always true

424 return 0 

425 await self._post_response(projection, thread_id, branch=None) 

426 return thread_id 

427 elif isinstance(projection, A2AReject):

429 if turn is None or projection.tool_call_id not in turn.branches: 

430 return 0 

431 return await self._resolve_branch( 

432 projection.correlation_id, projection.tool_call_id, "rejected", note=projection.text

449 async def _open_branch(self, request: A2ARequest) -> int: 

450 channel_id = await self._channel() 

451 turn = self._turns.setdefault( 

452 request.correlation_id, 

453 _Turn(root_agent=request.caller, subject=request.message), 

454 ) 

455 sent = await self._personas.send_components( 

456 persona=persona_for(request.caller), 

457 channel_id=channel_id, 

458 view=_ConsultCard(request, state="pending"),

460 ) 

461 turn.branches[request.tool_call_id] = _Branch( 

462 request=request, 

463 message_id=sent.id, 

464 opened_at=time.monotonic(), 

465 ) 

466 if turn.thread_id is None: 

467 turn.thread_id = await self._resolver.create_anchored_thread( 

468 channel_id,

471 )

509 async def _post_response(self, projection: A2AReply, thread_id: int | None, *, branch: _Branch | None) -> None: 

510 await self._steps.flush(projection.correlation_id) 

511 channel_id = self._channel_id or await self._channel() 

512 prefix = f"-# **↩ {_card_plain(projection.peer)} → {_card_plain(projection.caller)} · response**" 

513 content = projection.text or _EMPTY_PLACEHOLDER 

514 chunks = chunk_split(f"{prefix}\n\n{content}") 

515 for chunk in chunks:

518 channel_id=channel_id, 

519 content=chunk, 

520 thread_id=thread_id, 

521 suppress_mentions=True,

524 # if a later chunk fails. A first-send failure leaves the card pending 

525 # so finish can honestly correct it. 

526 if branch is not None: 526 ↛ 515line 526 didn't jump to line 515 because the condition on line 526 was always true

527 branch.state = "replied"
