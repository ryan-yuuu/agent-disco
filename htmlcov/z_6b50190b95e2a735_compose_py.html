1"""Render the Process Compose project that supervises a calfcord host's SUBSTRATE. 

2 

3The Process Compose YAML is *derived state* — calfcord generates it from config; 

4the user never edits it (design §3.1). This module is the pure heart of that 

5generation: home dir + launcher prefix in, a process-compose project ``dict`` 

6out. No filesystem, no broker, no network — so the structure is fully 

7golden-testable and the same generator works whether the host runs detached, in 

8dev, or under a future supervisor swap. 

9 

10**Phase 2 — the substrate is all that lives here now.** Process Compose cannot 

11hot-add a process to a live project (a ``POST /project`` bounces every PID — 

12proven on the pinned and latest builds), so the roster (agents, ``tools``, and 

13each ``mcp-<server>``) moved OFF Process Compose and is now a detached process 

14per slot, spawned via :mod:`procspawn` with a pidfile under ``state/run``. Only 

15the **substrate** — ``broker`` (when compose-managed) and ``bridge`` — is 

16declared here, autostarting under ``disco start``. 

17 

18The shape is the §13.2 Phase-0 contract, pinned against Process Compose 

19``v1.110.0`` (config schema ``version: "0.5"``): 

20 

21* Every ``command`` invokes the calfcord *launcher* (the shim) rather than a 

22 reconstructed ``uv run`` line, so the venv + ``--env-file`` + default env come 

23 from one place and no secret literal is ever inlined into the YAML (design 

24 §12.3). The launcher prefix is a parameter so this generator is mode-agnostic 

25 and unit-testable. 

26* ``bridge`` gates on the broker via ``depends_on`` (``process_healthy``); 

27 readiness is an ``exec`` probe (the bridge has no HTTP server) calling 

28 ``<launcher> _healthcheck <component>``. 

29* ``restart: always`` for the substrate (``broker``, ``bridge``): both exit 0 on 

30 a clean signal-less return, so ``on_failure`` would never fire to recover an 

31 uncommanded clean exit — they must ``always`` restart. Never ``exit_on_failure``. 

32 

33The REST port is intentionally absent: it is a flag to ``process-compose up`` 

34(``-p <PC_PORT>``), not a field in the project file (design §13.2). 

35"""

77# The filename stem of the supervisor's *own* log (``process-compose up -L 

78# <stem>.log``). It is not a process the generator declares, but it sits beside 

79# the per-process logs and is a legitimate tail target, so its name must agree 

80# across the modules that write it (``lifecycle._SUPERVISOR_LOG_FILENAME``) and 

81# read it (``cli.logs``). Both derive from this one stem so the literal can never 

82# drift: lifecycle reconstructs the filename as ``SUPERVISOR_LOG_STEM + ".log"``, 

83# and ``cli.logs`` passes the stem to ``_log_location`` (which appends ``.log``). 

84# Homed here because both consumers already import ``compose`` and ``compose`` is 

85# import-light, keeping the logs CLI's decoupling intact. 

86SUPERVISOR_LOG_STEM = "process-compose" 

87 

88# Slot names owned by the substrate (``broker``/``bridge``, on Process Compose), 

89# the ``tools`` singleton (a roster slot spawned off PC), and the supervisor's 

90# own log stem (``process-compose`` — not a slot, but every roster slot logs to 

91# ``state/logs/<slot>.log``, and the supervisor's ``-L`` log is 

92# ``state/logs/process-compose.log``, so an agent by that name would share and 

93# rotate-at-spawn the live supervisor's log). An agent whose id equalled one of 

94# these would collide with that process/file, so agent names may never take 

95# them: the create-/parse-time guard lives in :data:`calfcord.agents.identifier 

96# .RESERVED_AGENT_IDS` (this module cannot import it — ``calfcord.agents``'s 

97# package init pulls calfkit, and the supervisor stays import-light — so the two 

98# literals are pinned equal by test instead: ``test_identifier``). 

99_RESERVED_PROCESS_NAMES = frozenset({"broker", "bridge", "tools", SUPERVISOR_LOG_STEM}) 

100 

101# The slot-name convention for MCP servers, homed here (like SUPERVISOR_LOG_STEM) 

102# because the roster surfaces spawn ``mcp-<server>`` slots and the shared 

103# ``_workspace`` scan classifies them — all must agree on the literal, and compose 

104# is the import-light module the supervisor package already leans on. The same 

105# literal is a reserved agent-name prefix (:data:`calfcord.agents.identifier 

106# .MCP_SLOT_PREFIX`); pinned equal by test for the same import-light reason. 

107MCP_SLOT_PREFIX = "mcp-"

117# Hosts whose broker URL means "a local broker calfcord itself supervises": the 

118# native install runs Tansu as the compose-managed ``broker`` process bound to 

119# loopback. Anything else is an EXTERNAL broker the operator runs elsewhere, so 

120# calfcord must neither declare a local broker slot for it nor start one. The 

121# empty host (":9092" / bare "") is treated as loopback (no host == localhost). 

122_LOOPBACK_HOSTS = frozenset({"", "localhost", "::1"})

136 return url[1:end]

142def _is_loopback_host(host: str) -> bool:

148def broker_is_compose_managed(server_urls: str) -> bool: 

149 """Whether ``server_urls`` designates a local broker calfcord supervises. 

150 

151 This is the single predicate that keeps two decisions from diverging (design 

152 §13.2): whether the rendered project declares a local ``broker`` process, and 

153 whether :func:`lifecycle.start` runs its pre-launch broker fast-fail probe. 

154 

155 A **loopback** URL (``localhost`` / ``127.0.0.0/8`` / ``::1``, with or without 

156 a port) means calfcord runs Tansu as the compose-managed ``broker`` process: 

157 ``process-compose up`` launches it (with its own readiness probe + depends_on 

158 graph), so `start` must NOT pre-probe it — the broker is cold until `up` starts 

159 it. A non-loopback (external) URL means the operator's broker lives elsewhere, 

160 so no local slot is declared and the pre-launch fast-fail stands. A 

161 comma-separated bootstrap list is compose-managed only if EVERY host is 

162 loopback; an empty string is external (unknown → keep the fast-fail). 

163 """ 

164 hosts = [h for h in (part.strip() for part in server_urls.split(",")) if h] 

165 if not hosts: 

166 return False 

167 return all(_is_loopback_host(_host_of(h)) for h in hosts)

224def build_compose_project( 

225 *, 

226 home: str, 

227 launcher: str, 

228 broker_managed: bool = True, 

229) -> dict: 

230 """Build the Process Compose project that supervises one calfcord host's SUBSTRATE. 

231 

232 Only the substrate lives on Process Compose now (Phase 2): ``broker`` (when 

233 compose-managed) and ``bridge``. The roster — agents, ``tools``, and each 

234 ``mcp-<server>`` — is spawned off PC as detached processes (see :mod:`procspawn` 

235 and the ``roster`` / ``component`` / ``mcp_roster`` surfaces), so it is no 

236 longer declared here. 

237 

238 ``home`` is ``$CALFCORD_HOME`` — only the per-process ``state/logs/<name>.log`` 

239 paths use it. ``launcher`` is the shim prefix every ``command`` is built on 

240 (e.g. ``$CALFCORD_HOME/shims/disco``); the generator never reconstructs 

241 ``uv run`` flags or inlines secrets. 

242 

243 ``broker_managed`` (default ``True``, the native local install) declares Tansu 

244 as the autostart ``broker`` process and health-gates the bridge on it. Set 

245 ``False`` for an EXTERNAL broker (the operator runs Kafka elsewhere): no local 

246 ``broker`` process is declared — starting an ephemeral broker nobody talks to is 

247 wrong — and the bridge carries no ``depends_on`` the broker (a dependency on an 

248 undeclared process would make process-compose reject the project). The caller 

249 derives this flag from :func:`broker_is_compose_managed` so manifest inclusion 

250 and `start`'s pre-launch probe cannot diverge (design §13.2). 

251 

252 Returns a plain ``dict`` (serialize with :func:`render_compose`). 

253 """ 

254 processes: dict[str, dict] = {} 

255 

256 # The broker gate the bridge shares — present only when the broker is a local 

257 # compose process. For an external broker there is no local ``broker`` process 

258 # to health-gate on, so the dependency is dropped entirely (process-compose 

259 # rejects a depends_on to an undeclared process). 

260 broker_dep = {"broker": _HEALTHY} if broker_managed else None 

261 

262 # Substrate — autostarts, health-gated. The broker (when compose-managed) has a 

263 # readiness probe checking metadata reachability (not bare TCP); the bridge's 

264 # checks the Discord heartbeat. `start` gates on the bridge. 

265 if broker_managed: 

266 processes["broker"] = _process( 

267 command=f"{launcher} broker", 

268 home=home, 

269 name="broker", 

270 disabled=False, 

271 restart_policy="always", 

272 readiness_probe=_readiness_probe(launcher, "broker"), 

273 ) 

274 processes["bridge"] = _process( 

275 command=f"{launcher} run bridge", 

276 home=home, 

277 name="bridge", 

278 disabled=False, 

279 restart_policy="always", 

280 depends_on=broker_dep, 

281 readiness_probe=_readiness_probe(launcher, "bridge"), 

282 ) 

283 

284 return { 

285 "version": COMPOSE_SCHEMA_VERSION, 

286 "log_configuration": { 

287 "rotation": { 

288 "max_size_mb": _LOG_MAX_SIZE_MB, 

289 "max_age_days": _LOG_MAX_AGE_DAYS, 

290 "max_backups": _LOG_MAX_BACKUPS, 

291 "compress": _LOG_COMPRESS, 

292 } 

293 }, 

294 "processes": processes, 

295 }

298def render_compose( 

299 *, 

300 home: str, 

301 launcher: str, 

302 broker_managed: bool = True, 

303) -> str: 

304 """Render the Process Compose SUBSTRATE project as a YAML string. 

305 

306 Thin serializer over :func:`build_compose_project` — ``sort_keys=False`` keeps 

307 the broker-before-bridge ordering the builder emits, which makes the generated 

308 file readable even though the user never edits it. ``broker_managed`` is 

309 threaded through unchanged (see :func:`build_compose_project`). 

310 """ 

311 project = build_compose_project( 

312 home=home, 

313 launcher=launcher, 

314 broker_managed=broker_managed, 

315 ) 

316 return yaml.safe_dump(project, sort_keys=False)
